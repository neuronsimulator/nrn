52  /*
53  treesolver
54 
55  a - above the diagonal
56  d - diagonal
57  b - below the diagonal
58  rhs - right hand side, which is changed to the result
59  pindex - parent indices
60  n - number of states
61  */
62 
63  int i;
64 
65  /* triang */
66  for (i = n - 1; i > 0; --i) {
67  int pin = pindex[i];

69  double p;
70  p = a[i] / d[i];
