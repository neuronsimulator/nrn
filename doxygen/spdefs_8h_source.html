307 /* Complex division: to = num / den */
308 #define CMPLX_DIV(to, num, den) \
309  { \
310  RealNumber r_, s_; \
311  if (((den).Real >= (den).Imag AND(den).Real > -(den).Imag) OR((den).Real < (den).Imag AND(den).Real <= -(den).Imag)) { \
312  r_ = (den).Imag / (den).Real; \
313  s_ = (den).Real + r_ * (den).Imag; \
314  (to).Real = ((num).Real + r_ * (num).Imag) / s_; \
315  (to).Imag = ((num).Imag - r_ * (num).Real) / s_; \
316  } else { \
317  r_ = (den).Real / (den).Imag; \
318  s_ = (den).Imag + r_ * (den).Real; \
319  (to).Real = (r_ * (num).Real + (num).Imag) / s_; \
320  (to).Imag = (r_ * (num).Imag - (num).Real) / s_; \
321  } \
322  }
323 
324 /* Complex division and assignment: num /= den */
325 #define CMPLX_DIV_ASSIGN(num, den) \
326  { \
327  RealNumber r_, s_, t_; \
328  if (((den).Real >= (den).Imag AND(den).Real > -(den).Imag) OR((den).Real < (den).Imag AND(den).Real <= -(den).Imag)) { \
329  r_ = (den).Imag / (den).Real; \
330  s_ = (den).Real + r_ * (den).Imag; \
331  t_ = ((num).Real + r_ * (num).Imag) / s_; \
332  (num).Imag = ((num).Imag - r_ * (num).Real) / s_; \
333  (num).Real = t_; \
334  } else { \
335  r_ = (den).Real / (den).Imag; \
336  s_ = (den).Imag + r_ * (den).Real; \
337  t_ = (r_ * (num).Real + (num).Imag) / s_; \
338  (num).Imag = (r_ * (num).Imag - (num).Real) / s_; \
339  (num).Real = t_; \
340  } \
341  }

444 /*
445  * MATRIX ELEMENT DATA STRUCTURE
446  *
447  * Every nonzero element in the matrix is stored in a dynamically allocated
448  * MatrixElement structure. These structures are linked together in an
449  * orthogonal linked list. Two different MatrixElement structures exist.
450  * One is used when only real matrices are expected, it is missing an entry
451  * for imaginary data. The other is used if complex matrices are expected.
452  * It contains an entry for imaginary data.
453  *
454  * >>> Structure fields:
455  * Real (RealNumber)
456  * The real portion of the value of the element. Real must be the first
457  * field in this structure.
458  * Imag (RealNumber)
459  * The imaginary portion of the value of the element. If the matrix
460  * routines are not compiled to handle complex matrices, then this
461  * field does not exist. If it exists, it must follow immediately after
462  * Real.
463  * Row (int)
464  * The row number of the element.
465  * Col (int)
466  * The column number of the element.
467  * NextInRow (struct MatrixElement *)
468  * NextInRow contains a pointer to the next element in the row to the
469  * right of this element. If this element is the last nonzero in the
470  * row then NextInRow contains NULL.
471  * NextInCol (struct MatrixElement *)
472  * NextInCol contains a pointer to the next element in the column below
473  * this element. If this element is the last nonzero in the column then
474  * NextInCol contains NULL.
475  * pInitInfo (char *)
476  * Pointer to user data used for initialization of the matrix element.
477  * Initialized to NULL.
478  *
479  * >>> Type definitions:
480  * ElementPtr
481  * A pointer to a MatrixElement.
482  * ArrayOfElementPtrs
483  * An array of ElementPtrs. Used for FirstInRow, FirstInCol and
484  * Diag pointer arrays.
485  */
486 
487 /* Begin `MatrixElement'. */
488 
489 struct MatrixElement {
490  RealNumber Real;
491 #if spCOMPLEX
492  RealNumber Imag;
493 #endif
494  int Row;
495  int Col;
496  struct MatrixElement* NextInRow;
497  struct MatrixElement* NextInCol;
498 #if INITIALIZE
499  char* pInitInfo;
500 #endif
501 };

556 /*
557  * MATRIX FRAME DATA STRUCTURE
558  *
559  * This structure contains all the pointers that support the orthogonal
560  * linked list that contains the matrix elements. Also included in this
561  * structure are other numbers and pointers that are used globally by the
562  * sparse matrix routines and are associated with one particular matrix.
563  *
564  * >>> Type definitions:
565  * MatrixPtr
566  * A pointer to MatrixFrame. Essentially, a pointer to the matrix.
567  *
568  * >>> Structure fields:
569  * AbsThreshold (RealNumber)
570  * The absolute magnitude an element must have to be considered as a
571  * pivot candidate, except as a last resort.
572  * AllocatedExtSize (int)
573  * The allocated size of the arrays used to translate external row and
574  * column numbers to their internal values.
575  * AllocatedSize (int)
576  * The currently allocated size of the matrix; the size the matrix can
577  * grow to when EXPANDABLE is set true and AllocatedSize is the largest
578  * the matrix can get without requiring that the matrix frame be
579  * reallocated.
580  * Complex (BOOLEAN)
581  * The flag which indicates whether the matrix is complex (true) or
582  * real.
583  * CurrentSize (int)
584  * This number is used during the building of the matrix when the
585  * TRANSLATE option is set true. It indicates the number of internal
586  * rows and columns that have elements in them.
587  * Diag (ArrayOfElementPtrs)
588  * Array of pointers that points to the diagonal elements.
589  * DoCmplxDirect (BOOLEAN *)
590  * Array of flags, one for each column in matrix. If a flag is true
591  * then corresponding column in a complex matrix should be eliminated
592  * in spFactor() using direct addressing (rather than indirect
593  * addressing).
594  * DoRealDirect (BOOLEAN *)
595  * Array of flags, one for each column in matrix. If a flag is true
596  * then corresponding column in a real matrix should be eliminated
597  * in spFactor() using direct addressing (rather than indirect
598  * addressing).
599  * Elements (int)
600  * The number of original elements (total elements minus fill ins)
601  * present in matrix.
602  * Error (int)
603  * The error status of the sparse matrix package.
604  * ExtSize (int)
605  * The value of the largest external row or column number encountered.
606  * ExtToIntColMap (int [])
607  * An array that is used to convert external columns number to internal
608  * external column numbers. Present only if TRANSLATE option is set true.
609  * ExtToIntRowMap (int [])
610  * An array that is used to convert external row numbers to internal
611  * external row numbers. Present only if TRANSLATE option is set true.
612  * Factored (BOOLEAN)
613  * Indicates if matrix has been factored. This flag is set true in
614  * spFactor() and spOrderAndFactor() and set false in spCreate()
