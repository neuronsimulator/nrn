276 static int depend_append(int idep, int* dependencies, int deptype, int type) {
277  /* append only if not already in dependencies and != type*/
278  bool add = true;
279  if (deptype == type) {
280  return idep;
281  }
282  for (int i = 0; i < idep; ++i) {
283  if (deptype == dependencies[i]) {
284  add = false;
285  break;
286  }
287  }
288  if (add) {
289  dependencies[idep++] = deptype;
290  }
291  return idep;
292 }

294 /* return list of types that this type depends on (10 should be more than enough) */
295 /* dependencies must be an array that is large enough to hold that array */
296 /* number of dependencies is returned */
297 int nrn_mech_depend(int type, int* dependencies) {
298  int dpsize = corenrn.get_prop_dparam_size()[type];
299  int* ds = corenrn.get_memb_func(type).dparam_semantics;
300  int idep = 0;
301  if (ds)
302  for (int i = 0; i < dpsize; ++i) {
303  if (ds[i] > 0 && ds[i] < 1000) {
304  int deptype = ds[i];
305  int idepnew = depend_append(idep, dependencies, deptype, type);
306  if ((idepnew > idep) && !corenrn.get_ion_write_dependency().empty() &&
307  !corenrn.get_ion_write_dependency()[deptype].empty()) {
308  auto& iwd = corenrn.get_ion_write_dependency()[deptype];
309  int size = iwd[0];
310  for (int j = 1; j < size; ++j) {
311  idepnew = depend_append(idepnew, dependencies, iwd[j], type);
312  }
313  }
314  idep = idepnew;
315  }
316  }
317  return idep;
318 }
