55 MLCG::reset()
56 {
57  int32_t seed1 = initialSeedOne;
58  int32_t seed2 = initialSeedTwo;
59 
60  // Most people pick stupid seed numbers that do not have enough
61  // bits. In this case, if they pick a small seed number, we
62  // map that to a specific seed.
63 
64  if (seed1 < 0) {
65  seed1 = (seed1 + 2147483561);
66  seed1 = (seed1 < 0) ? -seed1 : seed1;
67  }
68 
69  if (seed2 < 0) {
70  seed2 = (seed2 + 2147483561);
71  seed2 = (seed2 < 0) ? -seed2 : seed2;
72  }
73 
74  if (seed1 > -1 && seed1 < SEED_TABLE_SIZE) {
75  seedOne = seedTable[seed1];
76  } else {
77  seedOne = seed1 ^ seedTable[seed1 & (SEED_TABLE_SIZE-1)];
78  }
79 
80  if (seed2 > -1 && seed2 < SEED_TABLE_SIZE) {
81  seedTwo = seedTable[seed2];
82  } else {
83  seedTwo = seed2 ^ seedTable[ seed2 & (SEED_TABLE_SIZE-1) ];
84  }
85  seedOne = (seedOne % 2147483561) + 1;
86  seedTwo = (seedTwo % 2147483397) + 1;
