25 //
26 // This is an extension of the older implementation of Algorithm M
27 // which I previously supplied. The main difference between this
28 // version and the old code are:
29 //
30 // + Andres searched high & low for good constants for
31 // the LCG.
32 //
33 // + theres more bit chopping going on.
34 //
35 // The following contains his comments.
36 //
37 // agn@UNH.CS.CMU.EDU sez..
38 //
39 // The generator below is based on 2 well known
40 // methods: Linear Congruential (LCGs) and Additive
41 // Congruential generators (ACGs).
42 //
43 // The LCG produces the longest possible sequence
44 // of 32 bit random numbers, each being unique in
45 // that sequence (it has only 32 bits of state).
46 // It suffers from 2 problems: a) Independence
47 // isnt great, that is the (n+1)th number is
48 // somewhat related to the preceding one, unlike
49 // flipping a coin where knowing the past outcomes
50 // dont help to predict the next result. b)
51 // Taking parts of a LCG generated number can be
52 // quite non-random: for example, looking at only
53 // the least significant byte gives a permuted
54 // 8-bit counter (that has a period length of only
55 // 256). The advantage of an LCA is that it is
56 // perfectly uniform when run for the entire period
57 // length (and very uniform for smaller sequences
58 // too, if the parameters are chosen carefully).
59 //
60 // ACGs have extremly long period lengths and
61 // provide good independence. Unfortunately,
62 // uniformity isnt not too great. Furthermore, I
63 // didnt find any theoretically analysis of ACGs
64 // that addresses uniformity.
65 //
66 // The RNG given below will return numbers
67 // generated by an LCA that are permuted under
68 // control of a ACG. 2 permutations take place: the
69 // 4 bytes of one LCG generated number are
70 // subjected to one of 16 permutations selected by
71 // 4 bits of the ACG. The permutation a such that
72 // byte of the result may come from each byte of
73 // the LCG number. This effectively destroys the
74 // structure within a word. Finally, the sequence
75 // of such numbers is permuted within a range of
76 // 256 numbers. This greatly improves independence.
77 //
78 //
79 // Algorithm M as describes in Knuths "Art of Computer Programming",
80 // Vol 2. 1969
81 // is used with a linear congruential generator (to get a good uniform
82 // distribution) that is permuted with a Fibonacci additive congruential
83 // generator to get good independence.
84 //
85 // Bit, byte, and word distributions were extensively tested and pass
86 // Chi-squared test near perfect scores (>7E8 numbers tested, Uniformity
87 // assumption holds with probability > 0.999)
88 //
89 // Run-up tests for on 7E8 numbers confirm independence with
90 // probability > 0.97.
91 //
92 // Plotting random points in 2d reveals no apparent structure.
93 //
94 // Autocorrelation on sequences of 5E5 numbers (A(i) = SUM X(n)*X(n-i),
95 // i=1..512)
96 // results in no obvious structure (A(i) ~ const).
97 //
98 // Except for speed and memory requirements, this generator outperforms
99 // random() for all tests. (random() scored rather low on uniformity tests,
100 // while independence test differences were less dramatic).
101 //
102 // AGN would like to..
103 // thanks to M.Mauldin, H.Walker, J.Saxe and M.Molloy for inspiration & help.
104 //
105 // And I would (DGC) would like to thank Donald Kunth for AGN for letting me
106 // use his extensions in this implementation.
107 //

164 //
165 // The LCG used to scramble the ACG
166 //
167 //
168 // LC-parameter selection follows recommendations in
169 // "Handbook of Mathematical Functions" by Abramowitz & Stegun 10th, edi.
170 //
171 // LC_A = 251^2, ~= sqrt(2^32) = 66049
172 // LC_C = result of a long trial & error series = 3907864577
173 //
174 
175 static const uint32_t LC_A = 66049;
176 static const uint32_t LC_C = 3907864577U;
177 static inline uint32_t LCG(uint32_t x)
178 {
179  return( x * LC_A + LC_C );
180 }

183 ACG::ACG(uint32_t seed, int size)
184 {
185  int l;
186  initialSeed = seed;
187  //
188  // Determine the size of the state table
189  //
190 
191  for (l = 0;
192  randomStateTable[l][0] != -1 && randomStateTable[l][1] < size;
193  l++);
194 
195  if (randomStateTable[l][1] == -1) {
196  l--;
197  }
198 
199  initialTableEntry = l;
200 
201  stateSize = randomStateTable[ initialTableEntry ][ 1 ];
202  auxSize = randomStateTable[ initialTableEntry ][ 2 ];
203 
204  //
205  // Allocate the state table & the auxillary table in a single malloc
206  //
207 
208  state = new uint32_t[stateSize + auxSize];
209  auxState = &state[stateSize];
210 
211  reset();
212 }

214 //
215 // Initialize the state
216 //
217 void
218 ACG::reset()
219 {
220  uint32_t u;
221 
222  if (initialSeed < SEED_TABLE_SIZE) {
223  u = seedTable[ initialSeed ];
224  } else {
225  u = initialSeed ^ seedTable[ initialSeed & (SEED_TABLE_SIZE-1) ];
226  }
227 
228 
229  j = randomStateTable[ initialTableEntry ][ 0 ] - 1;
230  k = randomStateTable[ initialTableEntry ][ 1 ] - 1;
231 
232  int i;
233  for(i = 0; i < stateSize; i++) {
234  state[i] = u = LCG(u);
235  }
236 
237  for (i = 0; i < auxSize; i++) {
238  auxState[i] = u = LCG(u);
239  }
240 
241  k = u % stateSize;
242  int tailBehind = (stateSize - randomStateTable[ initialTableEntry ][ 0 ]);
243  j = k - tailBehind;
244  if (j < 0) {
245  j += stateSize;
246  }
247 
248  lcgRecurr = u;
249 
250  assert(sizeof(double) == 2 * sizeof(int32_t));
251 }

264 uint32_t
265 ACG::asLong()
266 {
267  uint32_t result = state[k] + state[j];
268  state[k] = result;
269  j = (j <= 0) ? (stateSize-1) : (j-1);
270  k = (k <= 0) ? (stateSize-1) : (k-1);
271 
272  short int auxIndex = (result >> 24) & (auxSize - 1);
273  uint32_t auxACG = auxState[auxIndex];
274  auxState[auxIndex] = lcgRecurr = LCG(lcgRecurr);
275 
276  //
277  // 3c is a magic number. We are doing four masks here, so we
278  // do not want to run off the end of the permutation table.
279  // This insures that we have always got four entries left.
280  //
281  uint32_t *perm = & randomPermutations[result & 0x3c];
282 
283  result = *(perm++) & auxACG;
284  result |= *(perm++) & ((auxACG << 24)
285  | ((auxACG >> 8)& 0xffffff));
286  result |= *(perm++) & ((auxACG << 16)
287  | ((auxACG >> 16) & 0xffff));
288  result |= *(perm++) & ((auxACG << 8)
289  | ((auxACG >> 24) & 0xff));
290 
291  return(result);
292 }
