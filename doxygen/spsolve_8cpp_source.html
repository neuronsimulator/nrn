62 /*
63  * SOLVE MATRIX EQUATION
64  *
65  * Performs forward elimination and back substitution to find the
66  * unknown vector from the RHS vector and factored matrix. This
67  * routine assumes that the pivots are associated with the lower
68  * triangular (L) matrix and that the diagonal of the upper triangular
69  * (U) matrix consists of ones. This routine arranges the computation
70  * in different way than is traditionally used in order to exploit the
71  * sparsity of the right-hand side. See the reference in spRevision.
72  *
73  * >>> Arguments:
74  * Matrix <input> (char *)
75  * Pointer to matrix.
76  * RHS <input> (RealVector)
77  * RHS is the input data array, the right hand side. This data is
78  * undisturbed and may be reused for other solves.
79  * Solution <output> (RealVector)
80  * Solution is the output data array. This routine is constructed such that
81  * RHS and Solution can be the same array.
82  * iRHS <optional input> (RealVector)
83  * iRHS is the imaginary portion of the input data array, the right
84  * hand side. This data is undisturbed and may be reused for other solves.
85  * This argument is only necessary if matrix is complex and if
86  * spSEPARATED_COMPLEX_VECTOR is set true.
87  * iSolution <optional output> (RealVector)
88  * iSolution is the imaginary portion of the output data array. This
89  * routine is constructed such that iRHS and iSolution can be
90  * the same array. This argument is only necessary if matrix is complex
91  * and if spSEPARATED_COMPLEX_VECTOR is set true.
92  *
93  * >>> Local variables:
94  * Intermediate (RealVector)
95  * Temporary storage for use in forward elimination and backward
96  * substitution. Commonly referred to as c, when the LU factorization
97  * equations are given as Ax = b, Lc = b, Ux = c Local version of
98  * Matrix->Intermediate, which was created during the initial
99  * factorization in function CreateInternalVectors() in the matrix
100  * factorization module.
101  * pElement (ElementPtr)
102  * Pointer used to address elements in both the lower and upper triangle
103  * matrices.
104  * pExtOrder (int *)
105  * Pointer used to sequentially access each entry in IntToExtRowMap
106  * and IntToExtColMap arrays. Used to quickly scramble and unscramble
107  * RHS and Solution to account for row and column interchanges.
108  * pPivot (ElementPtr)
109  * Pointer that points to current pivot or diagonal element.
110  * Size (int)
111  * Size of matrix. Made local to reduce indirection.
112  * Temp (RealNumber)
113  * Temporary storage for entries in arrays.
114  */

117 void spSolve(char* eMatrix, RealVector RHS, RealVector Solution, std::optional<RealVector> iRHS, std::optional<RealVector> iSolution)
118 {
119  MatrixPtr Matrix = (MatrixPtr)eMatrix;
120  ElementPtr pElement;
121  RealVector Intermediate;
122  RealNumber Temp;
123  int I, *pExtOrder, Size;
124  ElementPtr pPivot;
125 
126  /* Begin `spSolve'. */
127  ASSERT(IS_VALID(Matrix) AND IS_FACTORED(Matrix));
128 
129 #if spCOMPLEX
130  if (Matrix->Complex) {
131  SolveComplexMatrix(Matrix, RHS, Solution, iRHS, iSolution);
132  return;
133  }
134 #endif
135 
136 #if REAL
137  Intermediate = Matrix->Intermediate;
138  Size = Matrix->Size;
139 
140 /* Correct array pointers for ARRAY_OFFSET. */
141 #if NOT ARRAY_OFFSET
142  --RHS;
143  --Solution;
144 #endif
145 
146  /* Initialize Intermediate vector. */
147  pExtOrder = &Matrix->IntToExtRowMap[Size];
148  for (I = Size; I > 0; I--)
149  Intermediate[I] = RHS[*(pExtOrder--)];
150 
151  /* Forward elimination. Solves Lc = b.*/
152  for (I = 1; I <= Size; I++) {
153  /* This step of the elimination is skipped if Temp equals zero. */
154  if ((Temp = Intermediate[I]) != 0.0) {
155  pPivot = Matrix->Diag[I];
156  Intermediate[I] = (Temp *= pPivot->Real);
157 
158  pElement = pPivot->NextInCol;
159  while (pElement != NULL) {
160  Intermediate[pElement->Row] -= Temp * pElement->Real;
161  pElement = pElement->NextInCol;
162  }
163  }
164  }
165 
166  /* Backward Substitution. Solves Ux = c.*/
167  for (I = Size; I > 0; I--) {
168  Temp = Intermediate[I];
169  pElement = Matrix->Diag[I]->NextInRow;
170  while (pElement != NULL) {
171  Temp -= pElement->Real * Intermediate[pElement->Col];
172  pElement = pElement->NextInRow;
173  }
174  Intermediate[I] = Temp;
175  }
176 
177  /* Unscramble Intermediate vector while placing data in to Solution vector. */
178  pExtOrder = &Matrix->IntToExtColMap[Size];
179  for (I = Size; I > 0; I--)
180  Solution[*(pExtOrder--)] = Intermediate[I];
181 
182  return;
183 #endif /* REAL */
184 }

242 static void SolveComplexMatrix(MatrixPtr Matrix, RealVector RHS, RealVector Solution, std::optional<RealVector> iRHS, std::optional<RealVector> iSolution)
243 {
244  ElementPtr pElement;
245  ComplexVector Intermediate;
246  int I, *pExtOrder, Size;
247  ElementPtr pPivot;
248  ComplexVector ExtVector;
249  ComplexNumber Temp;
250 
251  /* Begin `SolveComplexMatrix'. */
252 
253  Size = Matrix->Size;
254  Intermediate = (ComplexVector)Matrix->Intermediate;
255 
256 /* Correct array pointers for ARRAY_OFFSET. */
257 #if NOT ARRAY_OFFSET
258 #if spSEPARATED_COMPLEX_VECTORS
259  --RHS;
260  --iRHS;
261  --Solution;
262  --iSolution;
263 #else
264  RHS -= 2;
265  Solution -= 2;
266 #endif
267 #endif
268 
269  /* Initialize Intermediate vector. */
270  pExtOrder = &Matrix->IntToExtRowMap[Size];
271 
272 #if spSEPARATED_COMPLEX_VECTORS
273  for (I = Size; I > 0; I--) {
274  Intermediate[I].Real = RHS[*(pExtOrder)];
275  ASSERT(iRHS.has_value());
276  Intermediate[I].Imag = iRHS.value()[*(pExtOrder--)];
277  }
278 #else
279  ExtVector = (ComplexVector)RHS;
280  for (I = Size; I > 0; I--)
281  Intermediate[I] = ExtVector[*(pExtOrder--)];
282 #endif
283 
284  /* Forward substitution. Solves Lc = b.*/
285  for (I = 1; I <= Size; I++) {
286  Temp = Intermediate[I];
287 
288  /* This step of the substitution is skipped if Temp equals zero. */
289  if ((Temp.Real != 0.0) OR(Temp.Imag != 0.0)) {
290  pPivot = Matrix->Diag[I];
291  /* Cmplx expr: Temp *= (1.0 / Pivot). */
292  CMPLX_MULT_ASSIGN(Temp, *pPivot);
293  Intermediate[I] = Temp;
294  pElement = pPivot->NextInCol;
295  while (pElement != NULL) {
296  /* Cmplx expr: Intermediate[Element->Row] -= Temp * *Element. */
297  CMPLX_MULT_SUBT_ASSIGN(Intermediate[pElement->Row],
298  Temp, *pElement);
299  pElement = pElement->NextInCol;
300  }
301  }
302  }
303 
304  /* Backward Substitution. Solves Ux = c.*/
305  for (I = Size; I > 0; I--) {
306  Temp = Intermediate[I];
307  pElement = Matrix->Diag[I]->NextInRow;
308 
309  while (pElement != NULL) {
310  /* Cmplx expr: Temp -= *Element * Intermediate[Element->Col]. */
311  CMPLX_MULT_SUBT_ASSIGN(Temp, *pElement, Intermediate[pElement->Col]);
312  pElement = pElement->NextInRow;
313  }
314  Intermediate[I] = Temp;
315  }
316 
317  /* Unscramble Intermediate vector while placing data in to Solution vector. */
318  pExtOrder = &Matrix->IntToExtColMap[Size];
319 
320 #if spSEPARATED_COMPLEX_VECTORS
321  for (I = Size; I > 0; I--) {
322  Solution[*(pExtOrder)] = Intermediate[I].Real;
323  iSolution.value()[*(pExtOrder--)] = Intermediate[I].Imag;
324  }
325 #else
326  ExtVector = (ComplexVector)Solution;
327  for (I = Size; I > 0; I--)
328  ExtVector[*(pExtOrder--)] = Intermediate[I];
329 #endif
330 
331  return;
332 }
333 #endif /* spCOMPLEX */
