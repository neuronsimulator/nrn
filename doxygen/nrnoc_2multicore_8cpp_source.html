7 be a good time to regularize the method of freeing, allocating, and
8 updating those arrays. To recapitulate the history, Node used to
9 be the structure that held direct values for v, area, d, rhs, etc.
10 That continued to hold for the cray vectorization project which
11 introduced v_node, v_parent, memb_list. Cache efficiency introduced
12 actual_v, actual_area, actual_d, etc and the Node started pointing
13 into those arrays. Additional nodes after allocation required updating
14 pointers to v and area since those arrays were freed and reallocated.
15 Now, the threads hold all these arrays and we want to update them
16 properly under the circumstances of changing topology, changing
17 number of threads, and changing distribution of cells on threads.
18 Note there are no longer global versions of any of these arrays.
19 We do not want to update merely due to a change in area. Recently
20 we have dealt with diam, area, ri on a section basis. We generally
21 desire an update just before a simulation when the efficient
22 structures are necessary. This is reasonably well handled by the
23 v_structure_change flag which historically freed and reallocated
24 v_node and v_parent and, just before this comment,
25 ended up setting the NrnThread tml. This makes most of the old
26 memb_list vestigial and we now got rid of it except for
27 the artificial cells (and it is possibly not really necessary there).
28 Switching between sparse and tree matrix just cause freeing and
29 reallocation of actual_rhs.

33 it. We have two major cases, call to pc.nthread and change in

125 void worker_main(worker_conf_t* my_wc_ptr,
126  std::condition_variable* my_cond_ptr,
127  std::mutex* my_mut_ptr) {
128  assert(my_cond_ptr);
129  assert(my_mut_ptr);
130  assert(my_wc_ptr);
131  auto& cond{*my_cond_ptr};
132  auto& mut{*my_mut_ptr};
133  auto& wc{*my_wc_ptr};
134  for (;;) {
135  if (busywait_) {
136  // WARNING: this branch has not been extensively tested after the
137  // std::thread migration.
138  while (wc.flag == worker_flag::wait) {
139  ;
140  }
141  if (wc.flag == worker_flag::exit) {
142  return;
143  }
144  assert(wc.flag == worker_flag::execute_job);
145  std::visit(worker_kernel{wc.thread_id}, wc.job);
146  wc.flag = worker_flag::wait;
147  wc.job = std::monostate{};
148  cond.notify_one();
149  } else {
150  worker_conf_t conf{};
151  {
152  // Wait until we have a job to execute or we have been told to
153  // shut down.
154  std::unique_lock<std::mutex> lock{mut};
155  cond.wait(lock, [&wc] { return wc.flag != worker_flag::wait; });
156  // Received instructions from the coordinator thread.
157  assert(wc.flag == worker_flag::execute_job || wc.flag == worker_flag::exit);
158  if (wc.flag == worker_flag::exit) {
159  return;
160  }
161  assert(wc.flag == worker_flag::execute_job);
162  // Save the workload + argument to local variables before
163  // releasing the mutex.
164  conf = wc;
165  }
166  // Execute the workload without keeping the mutex
167  std::visit(worker_kernel{conf.thread_id}, conf.job);
168  // Signal that the work is completed and this thread is becoming
169  // idle
170  {
171  std::lock_guard<std::mutex> _{mut};
172  // Make sure we don't accidentally overwrite an exit signal from
173  // the coordinating thread.
174  if (wc.flag == worker_flag::execute_job) {
175  assert(wc == conf);
176  wc.flag = worker_flag::wait;
177  wc.job = std::monostate{};
178  }
179  }
180  // Notify the coordinating thread.
181  cond.notify_one();
182  }
183  }

663  // ForAllSections(sec)
664  ITERATE(qsec, section_list) {
665  Section* sec = hocSEC(qsec);
666  sec->order = -1;
667  }

669  FOR_THREADS(_nt) {
670  /* roots of this thread */

672  inode = 0;
673  ITERATE(qsec, sl) {
674  sec = hocSEC(qsec);

677  sec->order = order;
678  ++order;
679  nd = sec->parentnode;

681  inode += 1;
682  }
683  /* all children of what is already in secorder */
684  for (isec = order - _nt->ncell; isec < order; ++isec) {
685  sec = secorder[isec];
686  /* to make it easy to fill in PreSyn.nt_*/
687  sec->prop->dparam[9] = {neuron::container::do_not_search, _nt};
688  for (j = 0; j < sec->nnode; ++j) {
689  nd = sec->pnode[j];
690  nd->_nt = _nt;
691  inode += 1;
692  }
693  for (ch = sec->child; ch; ch = ch->sibling) {
694  assert(ch->order == -1);
695  secorder[order] = ch;
696  ch->order = order;
697  ++order;
698  }
699  }
700  _nt->end = inode;
701  CACHELINE_CALLOC(_nt->_v_node, Node*, inode);
702  CACHELINE_CALLOC(_nt->_v_parent, Node*, inode);
703  CACHELINE_CALLOC(_nt->_v_parent_index, int, inode);
704  }
705  /* do it again and fill _v_node and _v_parent */
706  /* index each cell section in relative order. Do offset later */
707  // ForAllSections(sec)
708  ITERATE(qsec, section_list) {
709  Section* sec = hocSEC(qsec);
710  sec->order = -1;
711  }
712  order = 0;
713  FOR_THREADS(_nt) {
714  /* roots of this thread */
715  sl = _nt->roots;
716  inode = 0;
717  ITERATE(qsec, sl) {
718  sec = hocSEC(qsec);
719  assert(sec->order == -1);
720  secorder[order] = sec;
721  sec->order = order;
722  ++order;
723  nd = sec->parentnode;
724  nd->_nt = _nt;
725  _nt->_v_node[inode] = nd;
726  _nt->_v_parent[inode] = nullptr; // because this is a root node
727  _nt->_v_node[inode]->v_node_index = inode;
728  ++inode;
729  }
730  /* all children of what is already in secorder */
731  for (isec = order - _nt->ncell; isec < order; ++isec) {
732  sec = secorder[isec];
733  /* to make it easy to fill in PreSyn.nt_*/
734  sec->prop->dparam[9] = {neuron::container::do_not_search, _nt};
735  for (j = 0; j < sec->nnode; ++j) {
736  nd = sec->pnode[j];
737  nd->_nt = _nt;
738  _nt->_v_node[inode] = nd;
739  if (j) {
740  _nt->_v_parent[inode] = sec->pnode[j - 1];
741  } else {
742  _nt->_v_parent[inode] = sec->parentnode;
743  }
744  _nt->_v_node[inode]->v_node_index = inode;
745  inode += 1;
746  }
747  for (ch = sec->child; ch; ch = ch->sibling) {
748  assert(ch->order == -1);
749  secorder[order] = ch;
750  ch->order = order;
751  ++order;
752  }
753  }
754  _nt->end = inode;
755  }
756  assert(order == section_count);
757  /*assert(inode == v_node_count);*/
758  /* not missing any */
759  // ForAllSections(sec)
760  ITERATE(qsec, section_list) {
761  Section* sec = hocSEC(qsec);
762  assert(sec->order != -1);
763  }

768  */

770  _nt->_v_node[inode]->_classical_parent = _nt->_v_parent[inode];
771  }
772  if (nrn_multisplit_setup_) {
773  /* classical order abandoned */
774  (*nrn_multisplit_setup_)();
775  }
776  /* because the d,rhs changed, if multisplit is used we need to update

780  nrn_multisplit_ptr_update();
