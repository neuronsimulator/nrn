65 static std::string check_output(std::string command) {
66  std::FILE* const p = popen(command.c_str(), "r");
67  if (!p) {
68  throw std::runtime_error("popen(" + command + ", \"r\") failed");
69  }
70  std::string output;
71  std::array<char, 1024> buffer{};
72  while (std::fgets(buffer.data(), buffer.size() - 1, p)) {
73  output += buffer.data();
74  }

77  err << "'" << command << "' did not terminate cleanly, pclose returned non-zero (" << code

79  << output;
80  throw std::runtime_error(err.str());
81  }
82  return output;
83 }

94 /**
95  * @brief Figure out which Python to load.
96  *
97  * When dynamic Python support is enabled, NEURON needs to figure out which
98  * libpythonX.Y to load, and then load it followed by the corresponding
99  * libnrnpythonX.Y. This can be steered both using commandline options and by
100  * using environment variables. The logic is as follows:
101  *
102  * * the -pyexe argument to nrniv (special) takes precedence over NRN_PYTHONEXE
103  * and we have to assume that the other NRN_PY* environment variables are not
104  * compatible with it and use nrnpyenv.sh to find compatible values for them,
105  * i.e. -pyexe implies that NRN_PY* are ignored.
106  * * if -pyexe is *not* passed, then we examine the NRN_PY* environment
107  * variables:
108  * * if all of them are set, nrnpyenv.sh is not run and they are assumed to
109  * form a coherent set
110  * * if only some, or none, of them are set, nrnpyenv.sh is run to fill in
111  * the missing values. NRN_PYTHONEXE is an input to nrnpyenv.sh, so if this
112  * is set then we will not search $PATH
113  */
114 static void set_nrnpylib() {
115  std::array<std::pair<std::string&, const char*>, 3> params{
116  {{nrnpy_pylib, "NRN_PYLIB"},
117  {nrnpy_pyexe, "NRN_PYTHONEXE"},
118  {nrnpy_pyversion, "NRN_PYTHONVERSION"}}};
119  auto const all_set = [&params]() {
120  return std::all_of(params.begin(), params.end(), [](auto const& p) {
121  return !p.first.empty();
122  });
123  };
124  if (nrnpy_pyexe.empty()) {
125  // -pyexe was not passed, read from the environment
126  for (auto& [glob_var, env_var]: params) {
127  if (const char* v = std::getenv(env_var)) {
128  glob_var = v;
129  }
130  }
131  if (all_set()) {
132  // the environment specified everything, nothing more to do
133  return;
134  }
135  }
136  // Populate missing values using nrnpyenv.sh. Pass the possibly-null value of nrnpy_pyexe, which
137  // may have come from -pyexe or NRN_PYTHONEXE, to nrnpyenv.sh. Do all of this on rank 0, and
138  // broadcast the results to other ranks afterwards.
139  if (nrnmpi_myid_world == 0) {
140  // Construct a command to execute
141  auto const command = []() -> std::string {
142 #ifdef MINGW
143  std::string bnrnhome{neuron_home}, fnrnhome{neuron_home};
144  std::replace(bnrnhome.begin(), bnrnhome.end(), '/', '\\');
145  std::replace(fnrnhome.begin(), fnrnhome.end(), '\\', '/');
146  return bnrnhome + R"(\mingw\usr\bin\bash )" + fnrnhome + "/bin/nrnpyenv.sh " +
147  nrnpy_pyexe + " --NEURON_HOME=" + fnrnhome;
148 #else
149  return "bash " + std::string{neuron_home} + "/../../bin/nrnpyenv.sh " + nrnpy_pyexe;
150 #endif
151  }();
152  // Execute the command, capture its stdout and wrap that in a C++ stream. This will throw if
153  // the commnand fails.
154  std::istringstream cmd_stdout{check_output(command)};
155  std::string line;
156  // if line is of the form:
157  // export FOO="bar"
158  // then proc_line(x, "FOO") sets x to bar
159  auto const proc_line = [](std::string_view line, auto& glob_var, std::string_view env_var) {
160  std::string_view const suffix{"\""};
161  auto const prefix = "export " + std::string{env_var} + "=\"";
162  if (starts_with(line, prefix) && ends_with(line, suffix)) {
163  line.remove_prefix(prefix.size());
164  line.remove_suffix(suffix.size());
165  if (!glob_var.empty() && glob_var != line) {
166  std::cout << "WARNING: overriding " << env_var << '=' << glob_var << " with "
167  << line << std::endl;
168  }
169  glob_var = line;
170  }
171  };
172  // Process the output of nrnpyenv.sh line by line
173  while (std::getline(cmd_stdout, line)) {
174  for (auto& [glob_var, env_var]: params) {
175  proc_line(line, glob_var, env_var);
176  }
177  }
178  // After having run nrnpyenv.sh, we should know everything about the Python library that is
179  // to be loaded.
180  if (!all_set()) {
181  std::ostringstream err;
182  err << "After running nrnpyenv.sh (" << command << ") with output:\n"
183  << cmd_stdout.str()
184  << "\nwe are still missing information about the Python to be loaded:\n"
185  << " nrnpy_pyexe=" << nrnpy_pyexe << '\n'
186  << " nrnpy_pylib=" << nrnpy_pylib << '\n'
187  << " nrnpy_pyversion=" << nrnpy_pyversion << '\n';
188  throw std::runtime_error(err.str());
189  }
190  }
191 #if NRNMPI
192  if (nrnmpi_numprocs_world > 1) { // 0 broadcasts to everyone else.
193  nrnmpi_str_broadcast_world(nrnpy_pyexe, 0);
194  nrnmpi_str_broadcast_world(nrnpy_pylib, 0);
195  nrnmpi_str_broadcast_world(nrnpy_pyversion, 0);
196  }
197 #endif
198 }
199 #endif

256  std::string pyversion{};
257  if (auto const pv10 = nrn_is_python_extension; pv10 > 0) {
258  // pv10 is one of the packed integers like 310 (3.10) or 38 (3.8)
259  auto const factor = (pv10 >= 100) ? 100 : 10;
260  pyversion = std::to_string(pv10 / factor) + "." + std::to_string(pv10 % factor);
261  } else {
262  if (nrnpy_pylib.empty() || nrnpy_pyversion.empty()) {
263  std::cerr << "Do not know what Python to load [nrnpy_pylib=" << nrnpy_pylib
264  << " nrnpy_pyversion=" << nrnpy_pyversion << ']' << std::endl;
265  return nullptr;
266  }
267  pyversion = nrnpy_pyversion;
268  // It's possible to get this far with an incompatible version, if nrnpy_pyversion and
269  // friends were set from the environment to bypass nrnpyenv.sh, and nrniv -python was
270  // launched.
271  auto const& supported_versions = neuron::config::supported_python_versions;
272  auto const iter =
273  std::find(supported_versions.begin(), supported_versions.end(), pyversion);
274  if (iter == supported_versions.end()) {
275  std::cerr << "Python " << pyversion
276  << " is not supported by this NEURON installation (supported:";
277  for (auto const& good_ver: supported_versions) {
278  std::cerr << ' ' << good_ver;
279  }

281  "NRN_PYLIB, NRN_PYTHONEXE and NRN_PYTHONVERSION settings that are "
282  "incompatible with this NEURON. Try unsetting them."
283  << std::endl;
284  return nullptr;

286  }

288  std::string name;

291  name.append(pyversion);

293 #ifndef MINGW

298  if (!handle) {
