32 data_handle<T> transform(data_handle<T> handle, Transform type) {
33  if (type == Transform::None) {

37  } else {
38  assert(type == Transform::ViaGenericDataHandle);
39  return static_cast<data_handle<T>>(generic_data_handle{handle});
