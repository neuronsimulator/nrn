390  switch (*pfmt) {

392  lflag += 1;
393  break;
394 
395  case 'd':
396  case 'o':
397  case 'x':
398  if (lflag) {

402  pfrag[-1] = 'l';

405  } else {

407  }

410 
411  case 'c':

413  didit = 1;
414  break;
415 
416  case 'f':
417  case 'e':
418  case 'g':

420  didit = 1;
421  break;
422

444  }

570  if always is 0 then
571  xopen only if file of that name not already loaded with one of
572  the load_xxx functions
573  and search in the current, $HOC_LIBRARY_PATH,
574  $NEURONHOME/lib/hoc directories (in that order) for

577  that it can xopen files relative to its location.
578  */

592  is_loaded = 0;
593  if (!loaded) {

595  }
596  ITERATE(q, loaded) {

598  if (!always) {
599  return 1;

601  is_loaded = 1;

606  /* maybe the name already has an explicit path */
607  expname[hoc_load_file_size_ - 1] = '\0';
608  strncpy(expname, expand_env_var(name), hoc_load_file_size_);
609  assert(expname[hoc_load_file_size_ - 1] == '\0');
610  name = expname;

612  strncpy(path, name, base - name);
613  path[base - name] = '\0';
614  ++base;
615  f = fopen(name, "r");
616  } else {

638  hlp = 0;
639  }

646  }
647  } else {
648  break;
649  }
650  }
651  }

655  nrn_assert(snprintf(fname, hoc_load_file_size_, "%s/%s", path, base) <

770  int size = 0;
771  char* p = NULL;
772 
773  if (!nrnpy_pr_stdoe_callback || (stream != stderr && stream != stdout)) {
774  size = vfprintf(stream, fmt, ap);
775  return size;
776  }
777 
778  /* Determine required size */
779  va_list apc;
780 #ifndef va_copy
781 #if defined(__GNUC__) || defined(__clang__)
782 #define va_copy(dest, src) __builtin_va_copy(dest, src)
783 #else
784 #define va_copy(dest, src) (dest = src)
785 #endif
786 #endif
787  va_copy(apc, ap);
788  size = vsnprintf(p, size, fmt, apc);
789  va_end(apc);
790

792  return 0;
793 
794  size++; /* For '\0' */
795  p = static_cast<char*>(malloc(size));
796  if (p == NULL)
797  return 0;
798 
799  size = vsnprintf(p, size, fmt, ap);
800  if (size < 0) {
801  free(p);
802  return 0;
803  }
804 
805  // if any non-ascii translate to '?' or nrnpy_pr will raise an exception.

810  }
