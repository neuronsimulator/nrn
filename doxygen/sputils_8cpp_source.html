85 /*
86  * PREORDER MODIFIED NODE ADMITTANCE MATRIX TO REMOVE ZEROS FROM DIAGONAL
87  *
88  * This routine massages modified node admittance matrices to remove
89  * zeros from the diagonal. It takes advantage of the fact that the
90  * row and column associated with a zero diagonal usually have
91  * structural ones placed symmetricly. This routine should be used
92  * only on modified node admittance matrices and should be executed
93  * after the matrix has been built but before the factorization
94  * begins. It should be executed for the initial factorization only
95  * and should be executed before the rows have been linked. Thus it
96  * should be run before using spScale(), spMultiply(),
97  * spDeleteRowAndCol(), or spNorm().
98  *
99  * This routine exploits the fact that the structural ones are placed
100  * in the matrix in symmetric twins. For example, the stamps for
101  * grounded and a floating voltage sources are
102  * grounded: floating:
103  * [ x x 1 ] [ x x 1 ]
104  * [ x x ] [ x x -1 ]
105  * [ 1 ] [ 1 -1 ]
106  * Notice for the grounded source, there is one set of twins, and for
107  * the floating, there are two sets. We remove the zero from the diagonal
108  * by swapping the rows associated with a set of twins. For example:
109  * grounded: floating 1: floating 2:
110  * [ 1 ] [ 1 -1 ] [ x x 1 ]
111  * [ x x ] [ x x -1 ] [ 1 -1 ]
112  * [ x x 1 ] [ x x 1 ] [ x x -1 ]
113  *
114  * It is important to deal with any zero diagonals that only have one
115  * set of twins before dealing with those that have more than one because
116  * swapping row destroys the symmetry of any twins in the rows being
117  * swapped, which may limit future moves. Consider
118  * [ x x 1 ]
119  * [ x x -1 1 ]
120  * [ 1 -1 ]
121  * [ 1 ]
122  * There is one set of twins for diagonal 4 and two for diagonal 3.
123  * Dealing with diagonal 4 first requires swapping rows 2 and 4.
124  * [ x x 1 ]
125  * [ 1 ]
126  * [ 1 -1 ]
127  * [ x x -1 1 ]
128  * We can now deal with diagonal 3 by swapping rows 1 and 3.
129  * [ 1 -1 ]
130  * [ 1 ]
131  * [ x x 1 ]
132  * [ x x -1 1 ]
133  * And we are done, there are no zeros left on the diagonal. However, if
134  * we originally dealt with diagonal 3 first, we could swap rows 2 and 3
135  * [ x x 1 ]
136  * [ 1 -1 ]
137  * [ x x -1 1 ]
138  * [ 1 ]
139  * Diagonal 4 no longer has a symmetric twin and we cannot continue.
140  *
141  * So we always take care of lone twins first. When none remain, we
142  * choose arbitrarily a set of twins for a diagonal with more than one set
143  * and swap the rows corresponding to that twin. We then deal with any
144  * lone twins that were created and repeat the procedure until no
145  * zero diagonals with symmetric twins remain.
146  *
147  * In this particular implementation, columns are swapped rather than rows.
148  * The algorithm used in this function was developed by Ken Kundert and
149  * Tom Quarles.
150  *
151  * >>> Arguments:
152  * eMatrix <input> (char *)
153  * Pointer to the matrix to be preordered.
154  *
155  * >>> Local variables;
156  * J (int)
157  * Column with zero diagonal being currently considered.
158  * pTwin1 (ElementPtr)
159  * Pointer to the twin found in the column belonging to the zero diagonal.
160  * pTwin2 (ElementPtr)
161  * Pointer to the twin found in the row belonging to the zero diagonal.
162  * belonging to the zero diagonal.
163  * AnotherPassNeeded (BOOLEAN)
164  * Flag indicating that at least one zero diagonal with symmetric twins
165  * remain.
166  * StartAt (int)
167  * Column number of first zero diagonal with symmetric twins.
168  * Swapped (BOOLEAN)
169  * Flag indicating that columns were swapped on this pass.
170  * Twins (int)
171  * Number of symmetric twins corresponding to current zero diagonal.
172  */
173 
174 void spMNA_Preorder(char* eMatrix)
175 {
176  MatrixPtr Matrix = (MatrixPtr)eMatrix;
177  int J, Size;
178  ElementPtr pTwin1 = 0, pTwin2 = 0;
179  int Twins, StartAt = 1;
180  BOOLEAN Swapped, AnotherPassNeeded;
181 
182  /* Begin `spMNA_Preorder'. */
183  ASSERT(IS_VALID(Matrix) AND NOT Matrix->Factored);
184 
185  if (Matrix->RowsLinked)
186  return;
187  Size = Matrix->Size;
188  Matrix->Reordered = YES;
189 
190  do {
191  AnotherPassNeeded = Swapped = NO;
192 
193  /* Search for zero diagonals with lone twins. */
194  for (J = StartAt; J <= Size; J++) {
195  if (Matrix->Diag[J] == NULL) {
196  Twins = CountTwins(Matrix, J, &pTwin1, &pTwin2);
197  if (Twins == 1) { /* Lone twins found, swap rows. */
198  SwapCols(Matrix, pTwin1, pTwin2);
199  Swapped = YES;
200  } else if ((Twins > 1) AND NOT AnotherPassNeeded) {
201  AnotherPassNeeded = YES;
202  StartAt = J;
203  }
204  }
205  }
206 
207  /* All lone twins are gone, look for zero diagonals with multiple twins. */
208  if (AnotherPassNeeded) {
209  for (J = StartAt; NOT Swapped AND(J <= Size); J++) {
210  if (Matrix->Diag[J] == NULL) {
211  Twins = CountTwins(Matrix, J, &pTwin1, &pTwin2);
212  SwapCols(Matrix, pTwin1, pTwin2);
213  Swapped = YES;
214  }
215  }
216  }
217  } while (AnotherPassNeeded);
218  return;
219 }

221 /*
222  * COUNT TWINS
223  *
224  * This function counts the number of symmetric twins associated with
225  * a zero diagonal and returns one set of twins if any exist. The
226  * count is terminated early at two.
227  */
228 
229 static int CountTwins(MatrixPtr Matrix, int Col, ElementPtr* ppTwin1, ElementPtr* ppTwin2)
230 {
231  int Row, Twins = 0;
232  ElementPtr pTwin1, pTwin2;
233 
234  /* Begin `CountTwins'. */
235 
236  pTwin1 = Matrix->FirstInCol[Col];
237  while (pTwin1 != NULL) {
238  if (ABS(pTwin1->Real) == 1.0) {
239  Row = pTwin1->Row;
240  pTwin2 = Matrix->FirstInCol[Row];
241  while ((pTwin2 != NULL) AND(pTwin2->Row != Col))
242  pTwin2 = pTwin2->NextInCol;
243  if ((pTwin2 != NULL) AND(ABS(pTwin2->Real) == 1.0)) { /* Found symmetric twins. */
244  if (++Twins >= 2)
245  return Twins;
246  (*ppTwin1 = pTwin1)->Col = Col;
247  (*ppTwin2 = pTwin2)->Col = Row;
248  }
249  }
250  pTwin1 = pTwin1->NextInCol;
251  }
252  return Twins;
253 }

283 /*
284  * SCALE MATRIX
285  *
286  * This function scales the matrix to enhance the possibility of
287  * finding a good pivoting order. Note that scaling enhances accuracy
288  * of the solution only if it affects the pivoting order, so it makes
289  * no sense to scale the matrix before spFactor(). If scaling is
290  * desired it should be done before spOrderAndFactor(). There
291  * are several things to take into account when choosing the scale
292  * factors. First, the scale factors are directly multiplied against
293  * the elements in the matrix. To prevent roundoff, each scale factor
294  * should be equal to an integer power of the number base of the
295  * machine. Since most machines operate in base two, scale factors
296  * should be a power of two. Second, the matrix should be scaled such
297  * that the matrix of element uncertainties is equilibrated. Third,
298  * this function multiplies the scale factors by the elements, so if
299  * one row tends to have uncertainties 1000 times smaller than the
300  * other rows, then its scale factor should be 1024, not 1/1024.
301  * Fourth, to save time, this function does not scale rows or columns
302  * if their scale factors are equal to one. Thus, the scale factors
303  * should be normalized to the most common scale factor. Rows and
304  * columns should be normalized separately. For example, if the size
305  * of the matrix is 100 and 10 rows tend to have uncertainties near
306  * 1e-6 and the remaining 90 have uncertainties near 1e-12, then the
307  * scale factor for the 10 should be 1/1,048,576 and the scale factors
308  * for the remaining 90 should be 1. Fifth, since this routine
309  * directly operates on the matrix, it is necessary to apply the scale
310  * factors to the RHS and Solution vectors. It may be easier to
311  * simply use spOrderAndFactor() on a scaled matrix to choose the
312  * pivoting order, and then throw away the matrix. Subsequent
313  * factorizations, performed with spFactor(), will not need to have
314  * the RHS and Solution vectors descaled. Lastly, this function
315  * should not be executed before the function spMNA_Preorder.
316  *
317  * >>> Arguments:
318  * eMatrix <input> (char *)
319  * Pointer to the matrix to be scaled.
320  * SolutionScaleFactors <input> (RealVector)
321  * The array of Solution scale factors. These factors scale the columns.
322  * All scale factors are real valued.
323  * RHS_ScaleFactors <input> (RealVector)
324  * The array of RHS scale factors. These factors scale the rows.
325  * All scale factors are real valued.
326  *
327  * >>> Local variables:
328  * lSize (int)
329  * Local version of the size of the matrix.
330  * pElement (ElementPtr)
331  * Pointer to an element in the matrix.
332  * pExtOrder (int *)
333  * Pointer into either IntToExtRowMap or IntToExtColMap vector. Used to
334  * compensate for any row or column swaps that have been performed.
335  * ScaleFactor (RealNumber)
336  * The scale factor being used on the current row or column.
337  */

830 /*
831  * CALCULATE DETERMINANT
832  *
833  * This routine in capable of calculating the determinant of the
834  * matrix once the LU factorization has been performed. Hence, only
835  * use this routine after spFactor() and before spClear().
836  * The determinant equals the product of all the diagonal elements of
837  * the lower triangular matrix L, except that this product may need

839  * determinant is determined by the number of row and column

851  * the actual determinant, Exponent should be added to the exponent of
852  * Determinant.

854  * The real portion of the determinant. This number is scaled to be
855  * greater than or equal to 1.0 and less than 10.0.
856  * piDeterminant <output> (RealNumber *)
857  * The imaginary portion of the determinant. When the matrix is real
858  * this pointer need not be supplied, nothing will be returned. This
859  * number is scaled to be greater than or equal to 1.0 and less than 10.0.
860  *
861  * >>> Local variables:
862  * Norm (RealNumber)
863  * L-infinity norm of a complex number.
864  * Size (int)

866  * Temp (RealNumber)
867  * Temporary storage for real portion of determinant.
868  */

870 void spDeterminant(char* eMatrix, int* pExponent, RealNumber* pDeterminant, std::optional<RealNumber*> piDeterminant)

874  RealNumber Norm, nr, ni;
875  ComplexNumber Pivot, cDeterminant;
876 
877 #define NORM(a) (nr = ABS((a).Real), ni = ABS((a).Imag), MAX(nr, ni))

879  /* Begin `spDeterminant'. */

881  *pExponent = 0;
882 
883  if (Matrix->Error == spSINGULAR) {
884  *pDeterminant = 0.0;
885 #if spCOMPLEX

896 
897 #if spCOMPLEX

901  cDeterminant.Imag = 0.0;
902

904  CMPLX_RECIPROCAL(Pivot, *Matrix->Diag[I]);
905  CMPLX_MULT_ASSIGN(cDeterminant, Pivot);
906

908  Norm = NORM(cDeterminant);
909  if (Norm != 0.0) {
910  while (Norm >= 1.0e12) {
911  cDeterminant.Real *= 1.0e-12;
912  cDeterminant.Imag *= 1.0e-12;
913  *pExponent += 12;
914  Norm = NORM(cDeterminant);
915  }
916  while (Norm < 1.0e-12) {
917  cDeterminant.Real *= 1.0e12;
918  cDeterminant.Imag *= 1.0e12;
919  *pExponent -= 12;
920  Norm = NORM(cDeterminant);
921  }
922  }
923  }
924 
925  /* Scale Determinant again, this time to be between 1.0 <= x < 10.0. */
926  Norm = NORM(cDeterminant);
927  if (Norm != 0.0) {
928  while (Norm >= 10.0) {
929  cDeterminant.Real *= 0.1;
930  cDeterminant.Imag *= 0.1;
931  (*pExponent)++;
932  Norm = NORM(cDeterminant);
933  }
934  while (Norm < 1.0) {
935  cDeterminant.Real *= 10.0;
936  cDeterminant.Imag *= 10.0;
937  (*pExponent)--;
938  Norm = NORM(cDeterminant);
939  }
940  }
941  if (Matrix->NumberOfInterchangesIsOdd)
942  CMPLX_NEGATE(cDeterminant);
943 
944  *pDeterminant = cDeterminant.Real;
945  *piDeterminant.value() = cDeterminant.Imag;
946  }
947 #endif /* spCOMPLEX */
948 #if REAL AND spCOMPLEX
949  else
950 #endif
951 #if REAL
952  { /* Real Case. */
953  *pDeterminant = 1.0;
954 
955  while (++I <= Size) {
956  *pDeterminant /= Matrix->Diag[I]->Real;
957 
958  /* Scale Determinant. */
959  if (*pDeterminant != 0.0) {
960  while (ABS(*pDeterminant) >= 1.0e12) {
961  *pDeterminant *= 1.0e-12;
962  *pExponent += 12;
963  }
964  while (ABS(*pDeterminant) < 1.0e-12) {
965  *pDeterminant *= 1.0e12;
966  *pExponent -= 12;
967  }
968  }
969  }
970 
971  /* Scale Determinant again, this time to be between 1.0 <= x < 10.0. */
972  if (*pDeterminant != 0.0) {
973  while (ABS(*pDeterminant) >= 10.0) {
974  *pDeterminant *= 0.1;
975  (*pExponent)++;
976  }
977  while (ABS(*pDeterminant) < 1.0) {
978  *pDeterminant *= 10.0;
979  (*pExponent)--;
980  }
981  }
982  if (Matrix->NumberOfInterchangesIsOdd)

1311 #define SLACK 1e4
1312 
1313  /* Begin `spCondition'. */
1314 
1315  ASSERT(IS_SPARSE(Matrix) AND IS_FACTORED(Matrix));

1317  if (Matrix->Error >= spFATAL)
1318  return 0.0;
1319  if (NormOfMatrix == 0.0) {
1320  *pError = spSINGULAR;
1321  return 0.0;
1322  }
1323 
1324 #if spCOMPLEX

1326  return ComplexCondition(Matrix, NormOfMatrix, pError);
1327 #endif

1329 #if REAL
1330  Size = Matrix->Size;
1331  T = Matrix->Intermediate;
1332 #if spCOMPLEX
1333  Tm = Matrix->Intermediate + Size;
1334 #else
1335  Tm = ALLOC(RealNumber, Size + 1);
1336  if (Tm == NULL) {
1337  *pError = spNO_MEMORY;
1338  return 0.0;
1339  }
1340 #endif
1341  for (I = Size; I > 0; I--)
1342  T[I] = 0.0;
1343 
1344  /*
1345  * Part 1. Ay = e.
1346  * Solve Ay = LUy = e where e consists of +1 and -1 terms with the sign
1347  * chosen to maximize the size of w in Lw = e. Since the terms in w can
1348  * get very large, scaling is used to avoid overflow.
1349  */
1350 
1351  /* Forward elimination. Solves Lw = e while choosing e. */
1352  E = 1.0;
1353  for (I = 1; I <= Size; I++) {
1354  pPivot = Matrix->Diag[I];
1355  if (T[I] < 0.0)
1356  Em = -E;
1357  else
1358  Em = E;
1359  Wm = (Em + T[I]) * pPivot->Real;
1360  if (ABS(Wm) > SLACK) {
1361  ScaleFactor = 1.0 / MAX(SQR(SLACK), ABS(Wm));
1362  for (K = Size; K > 0; K--)
1363  T[K] *= ScaleFactor;
1364  E *= ScaleFactor;
1365  Em *= ScaleFactor;
1366  Wm = (Em + T[I]) * pPivot->Real;
1367  }

1370  ASm = ABS(Em + T[I]);
1371

1373  pElement = pPivot->NextInCol;
1374  while (pElement != NULL) {

1376  Tm[Row] = T[Row] - (Wm * pElement->Real);
1377  T[Row] -= (Wp * pElement->Real);

1379  ASm += ABS(Tm[Row]);
1380  pElement = pElement->NextInCol;
1381  }

1383  /* If minus value causes more growth, overwrite T with its values. */
1384  if (ASm > ASp) {

1387  while (pElement != NULL) {

1389  pElement = pElement->NextInCol;
1390  }
1391  } else
1392  T[I] = Wp;
1393  }

1396  for (ASw = 0.0, I = Size; I > 0; I--)

1416  E *= ScaleFactor;
1417  }

1421  for (ASy = 0.0, I = Size; I > 0; I--)
1422  ASy += ABS(T[I]);
1423  ScaleFactor = 1.0 / (SLACK * ASy);
1424  if (ScaleFactor < 0.5) {
1425  for (I = Size; I > 0; I--)
1426  T[I] *= ScaleFactor;
1427  ASy = 1.0 / SLACK;
1428  E *= ScaleFactor;
1429  }

1431  /* Compute infinity-norm of T for O'Leary's estimate. */
1432  for (MaxY = 0.0, I = Size; I > 0; I--)
1433  if (MaxY < ABS(T[I]))
1434  MaxY = ABS(T[I]);

1436  /*
1437  * Part 2. A* z = y where the * represents the transpose.
1438  * Recall that A = LU implies A* = U* L*.
1439  */
1440 
1441  /* Forward elimination, U* v = y. */
1442  for (I = 1; I <= Size; I++) {
1443  pElement = Matrix->Diag[I]->NextInRow;
1444  while (pElement != NULL) {
1445  T[pElement->Col] -= T[I] * pElement->Real;
1446  pElement = pElement->NextInRow;
1447  }
1448  if (ABS(T[I]) > SLACK) {
1449  ScaleFactor = 1.0 / MAX(SQR(SLACK), ABS(T[I]));
1450  for (K = Size; K > 0; K--)
1451  T[K] *= ScaleFactor;
1452  ASy *= ScaleFactor;
1453  }
1454  }
1455 
1456  /* Compute 1-norm of T, which now contains v, and scale ||T|| to 1/SLACK. */
1457  for (ASv = 0.0, I = Size; I > 0; I--)
1458  ASv += ABS(T[I]);
1459  ScaleFactor = 1.0 / (SLACK * ASv);
1460  if (ScaleFactor < 0.5) {
1461  for (I = Size; I > 0; I--)
1462  T[I] *= ScaleFactor;
1463  ASy *= ScaleFactor;
1464  }
1465 
1466  /* Backward Substitution, L* z = v. */
1467  for (I = Size; I >= 1; I--) {
1468  pPivot = Matrix->Diag[I];
1469  pElement = pPivot->NextInCol;
1470  while (pElement != NULL) {
1471  T[I] -= pElement->Real * T[pElement->Row];
1472  pElement = pElement->NextInCol;
1473  }
1474  T[I] *= pPivot->Real;
1475  if (ABS(T[I]) > SLACK) {
1476  ScaleFactor = 1.0 / MAX(SQR(SLACK), ABS(T[I]));
1477  for (K = Size; K > 0; K--)
1478  T[K] *= ScaleFactor;
1479  ASy *= ScaleFactor;
1480  }
1481  }
1482 
1483  /* Compute 1-norm of T, which now contains z. */
1484  for (ASz = 0.0, I = Size; I > 0; I--)
1485  ASz += ABS(T[I]);
1486 
1487 #if NOT spCOMPLEX
1488  FREE(Tm);
1489 #endif
1490 
1491  Linpack = ASy / ASz;
1492  OLeary = E / MaxY;

1763 #if STABILITY
1764 /*
1765  * STABILITY OF FACTORIZATION
1766  *
1767  * The following routines are used to gauge the stability of a
1768  * factorization. If the factorization is determined to be too unstable,
1769  * then the matrix should be reordered. The routines compute quantities
1770  * that are needed in the computation of a bound on the error attributed
1771  * to any one element in the matrix during the factorization. In other
1772  * words, there is a matrix E = [e_ij] of error terms such that A+E = LU.
1773  * This routine finds a bound on |e_ij|. Erisman & Reid [1] showed that
1774  * |e_ij| < 3.01 u rho m_ij, where u is the machine rounding unit,
1775  * rho = max a_ij where the max is taken over every row i, column j, and
1776  * step k, and m_ij is the number of multiplications required in the
1777  * computation of l_ij if i > j or u_ij otherwise. Barlow [2] showed that
1778  * rho < max_i || l_i ||_p max_j || u_j ||_q where 1/p + 1/q = 1.
1779  *
1780  * The first routine finds the magnitude on the largest element in the
1781  * matrix. If the matrix has not yet been factored, the largest
1782  * element is found by direct search. If the matrix is factored, a
1783  * bound on the largest element in any of the reduced submatrices is
1784  * computed using Barlow with p = oo and q = 1. The ratio of these
1785  * two numbers is the growth, which can be used to determine if the
1786  * pivoting order is adequate. A large growth implies that
1787  * considerable error has been made in the factorization and that it
1788  * is probably a good idea to reorder the matrix. If a large growth
1789  * in encountered after using spFactor(), reconstruct the matrix and
1790  * refactor using spOrderAndFactor(). If a large growth is
1791  * encountered after using spOrderAndFactor(), refactor using
1792  * spOrderAndFactor() with the pivot threshold increased, say to 0.1.
1793  *
1794  * Using only the size of the matrix as an upper bound on m_ij and
1795  * Barlow's bound, the user can estimate the size of the matrix error
1796  * terms e_ij using the bound of Erisman and Reid. The second routine
1797  * computes a tighter bound (with more work) based on work by Gear
1798  * [3], |e_ij| < 1.01 u rho (t c^3 + (1 + t)c^2) where t is the
1799  * threshold and c is the maximum number of off-diagonal elements in
1800  * any row of L. The expensive part of computing this bound is
1801  * determining the maximum number of off-diagonals in L, which changes
1802  * only when the order of the matrix changes. This number is computed
1803  * and saved, and only recomputed if the matrix is reordered.
1804  *
1805  * [1] A. M. Erisman, J. K. Reid. Monitoring the stability of the
1806  * triangular factorization of a sparse matrix. Numerische
1807  * Mathematik. Vol. 22, No. 3, 1974, pp 183-186.
1808  *
1809  * [2] J. L. Barlow. A note on monitoring the stability of triangular
1810  * decomposition of sparse matrices. "SIAM Journal of Scientific
1811  * and Statistical Computing." Vol. 7, No. 1, January 1986, pp 166-168.
1812  *
1813  * [3] I. S. Duff, A. M. Erisman, J. K. Reid. "Direct Methods for Sparse
1814  * Matrices." Oxford 1986. pp 99.
1815  */
1816 
1817 /*
1818  * LARGEST ELEMENT IN MATRIX
1819  *
1820  * >>> Returns:
1821  * If matrix is not factored, returns the magnitude of the largest element in
1822  * the matrix. If the matrix is factored, a bound on the magnitude of the
1823  * largest element in any of the reduced submatrices is returned.
1824  *
1825  * >>> Arguments:
1826  * eMatrix <input> (char *)
1827  * Pointer to the matrix.
1828  */
1829 
1830 RealNumber spLargestElement(char* eMatrix)
1831 {
