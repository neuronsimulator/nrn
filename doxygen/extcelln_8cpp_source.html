349 void nrn_rhs_ext(NrnThread* _nt) {
350  int i, j, cnt;
351  Node *nd, *pnd, **ndlist;
352  Extnode *nde, *pnde;
353  Memb_list* ml = _nt->_ecell_memb_list;
354  if (!ml) {
355  return;
356  }
357  cnt = ml->nodecount;
358  ndlist = ml->nodelist;
359 
360  /* nd rhs contains -membrane current + stim current */
361  /* nde rhs contains stim current */
362  for (i = 0; i < cnt; ++i) {
363  nd = ndlist[i];
364  nde = nd->extnode;
365  *nde->_rhs[0] -= NODERHS(nd);
366 #if I_MEMBRANE
367  ml->data(i, sav_rhs_index) = *nde->_rhs[0];
368  /* and for daspk this is the ionic current which can be
369  combined later with i_cap before return from solve. */
370 #endif
371  }
372  for (i = 0; i < cnt; ++i) {
373  nd = ndlist[i];
374  nde = nd->extnode;
375  pnd = _nt->_v_parent[nd->v_node_index];
376  if (pnd) {
377  pnde = pnd->extnode;
378  /* axial contributions */
379  if (pnde) { /* parent sec may not be extracellular */
380  for (j = 0; j < nrn_nlayer_extracellular; ++j) {
381  double dv = pnde->v[j] - nde->v[j];
382  *nde->_rhs[j] -= nde->_b[j] * dv;
383  *pnde->_rhs[j] += nde->_a[j] * dv;
384  /* for the internal balance equation
385  take care of vi = vm + vx */
386  if (j == 0) {
387  NODERHS(nd) -= NODEB(nd) * dv;
388  NODERHS(pnd) += NODEA(nd) * dv;
389  }
390  }
391  } else { /* no extracellular in parent but still need
392  to take care of vi = vm + vx in this node.
393  Note that we are NOT grounding the parent
394  side of xraxial so equivalent to sealed.
395  Also note that when children of this node
396  do not have extracellular, we deal with
397  that, at end of this function. */
398  double dv = -nde->v[0];
399  NODERHS(nd) -= NODEB(nd) * dv;
400  NODERHS(pnd) += NODEA(nd) * dv;
401  }
402 
403  /* series resistance and battery to ground */
404  /* between nlayer-1 and ground */
405  j = nrn_nlayer_extracellular - 1;
406  *nde->_rhs[j] -= *nde->param[xg_index_ext(j)] *
407  (nde->v[j] - *nde->param[e_extracellular_index_ext()]);
408  for (--j; j >= 0; --j) { /* between j and j+1 layer */
409  double x = *nde->param[xg_index_ext(j)] * (nde->v[j] - nde->v[j + 1]);
410  *nde->_rhs[j] -= x;
411  *nde->_rhs[j + 1] += x;
412  }
413  }
414  }
415  cnt = _nt->_ecell_child_cnt;
416  for (i = 0; i < cnt; ++i) {
417  double dv;
418  nd = _nt->_ecell_children[i];
419  pnd = _nt->_v_parent[nd->v_node_index];
420  dv = pnd->extnode->v[0];
421  NODERHS(nd) -= NODEB(nd) * dv;
422  NODERHS(pnd) += NODEA(nd) * dv;
423  }
424 }

426 void nrn_setup_ext(NrnThread* _nt) {
427  int i, j, cnt;
428  Node *nd, *pnd, **ndlist;
429  double* pd;
430  double d, cfac, mfac;
431  Extnode *nde, *pnde;
432  Memb_list* ml = _nt->_ecell_memb_list;
433  if (!ml) {
434  return;
435  }
436  /*printnode("begin setup");*/
437  cnt = ml->nodecount;
438  ndlist = ml->nodelist;
439  cfac = .001 * _nt->cj;
440 
441  /* d contains all the membrane conductances (and capacitance) */
442  /* i.e. (cm/dt + di/dvm - dis/dvi)*[dvi] and
443  (dis/dvi)*[dvx] */
444  for (i = 0; i < cnt; ++i) {
445  nd = ndlist[i];
446  nde = nd->extnode;
447  d = NODED(nd);
448  /* nde->_d only has -ELECTRODE_CURRENT contribution */
449  d = (*nde->_d[0] += NODED(nd));
450  /* now d is only the membrane current contribution */
451  /* i.e. d = cm/dt + di/dvm */
452  *nde->_x12[0] -= d;
453  *nde->_x21[0] -= d;
454 #if I_MEMBRANE
455  ml->data(i, sav_g_index) = d;
456 #endif
457  }
458  /* series resistance, capacitance, and axial terms. */
459  for (i = 0; i < cnt; ++i) {
460  nd = ndlist[i];
461  nde = nd->extnode;
462  pnd = _nt->_v_parent[nd->v_node_index];
463  if (pnd) {
464  /* series resistance and capacitance to ground */
465  j = 0;
466  for (;;) { /* between j and j+1 layer */
467  mfac = (*nde->param[xg_index_ext(j)] + *nde->param[xc_index_ext(j)] * cfac);
468  *nde->_d[j] += mfac;
469  ++j;
470  if (j == nrn_nlayer_extracellular) {
471  break;
472  }
473  *nde->_d[j] += mfac;
474  *nde->_x12[j] -= mfac;
475  *nde->_x21[j] -= mfac;
476  }
477  pnde = pnd->extnode;
478  /* axial connections */
479  if (pnde) { /* parent sec may not be extracellular */
480  for (j = 0; j < nrn_nlayer_extracellular; ++j) {
481  *nde->_d[j] -= nde->_b[j];
482  *pnde->_d[j] -= nde->_a[j];
483  *nde->_a_matelm[j] += nde->_a[j];
484  *nde->_b_matelm[j] += nde->_b[j];
485  }
486  }
487  }
488  }
489  /*printnode("end setup_lhs");*/
490 }

493 void ext_con_coef(void) /* setup a and b */
494 {
495  int j, k;
496  double dx, area;
497  hoc_Item* qsec;
498  Node *nd, **pnd;
499  Extnode* nde;
500  double* pd;
501 
502  /* temporarily store half segment resistances in rhs */
503  // ForAllSections(sec)
504  ITERATE(qsec, section_list) {
505  Section* sec = hocSEC(qsec);
506  if (sec->pnode[0]->extnode) {
507  dx = section_length(sec) / ((double) (sec->nnode - 1));
508  for (j = 0; j < sec->nnode - 1; j++) {
509  nde = sec->pnode[j]->extnode;
510  for (k = 0; k < nrn_nlayer_extracellular; ++k) {
511  *nde->_rhs[k] = 1e-4 * *nde->param[xraxial_index_ext(k)] *
512  (dx / 2.); /*Megohms*/
513  }
514  }
515  /* last segment has 0 length. */
516  nde = sec->pnode[j]->extnode;
517  for (k = 0; k < nrn_nlayer_extracellular; ++k) {
518  *nde->_rhs[k] = 0.;
519  *nde->param[xc_index_ext(k)] = 0.;
520  *nde->param[xg_index_ext(k)] = 0.;
521  }
522  /* if owns a rootnode */
523  if (!sec->parentsec) {
524  nde = sec->parentnode->extnode;
525  for (k = 0; k < nrn_nlayer_extracellular; ++k) {
526  *nde->_rhs[k] = 0.;
527  *nde->param[xc_index_ext(k)] = 0.;
528  *nde->param[xg_index_ext(k)] = 0.;
529  }
530  }
531  }
532  }
533  /* assume that if only one connection at x=1, then they butte
534  together, if several connections at x=1
535  then last point is at x=1, has 0 area and other points are at
536  centers of nnode-1 segments.
537  If interior connection then child half
538  section connects straight to the point*/
539  /* for the near future we always have a last node at x=1 with
540  no properties */
541  // ForAllSections(sec)
542  ITERATE(qsec, section_list) {
543  Section* sec = hocSEC(qsec);
544  if (sec->pnode[0]->extnode) {
545  /* node half resistances in general get added to the
546  node and to the node's "child node in the same section".
547  child nodes in different sections don't involve parent
548  node's resistance */
549  nde = sec->pnode[0]->extnode;
550  for (k = 0; k < nlayer; ++k) {
551  nde->_b[k] = *nde->_rhs[k];
552  }
553  for (j = 1; j < sec->nnode; j++) {
554  nde = sec->pnode[j]->extnode;
555  for (k = 0; k < nlayer; ++k) {
556  nde->_b[k] = *nde->_rhs[k] + *(sec->pnode[j - 1]->extnode->_rhs[k]); /*megohms*/
557  }
558  }
559  }
560  }
561  // ForAllSections(sec)
562  ITERATE(qsec, section_list) {
563  Section* sec = hocSEC(qsec);
564  if (sec->pnode[0]->extnode) {
565  /* convert to siemens/cm^2 for all nodes except last
566  and microsiemens for last. This means that a*V = mamps/cm2
567  and a*v in last node = nanoamps. Note that last node
568  has no membrane properties and no area. It may perhaps recieve
569  current stimulus later */
570  /* first the effect of node on parent equation. Note That
571  last nodes have area = 1.e2 in dimensionless units so that
572  last nodes have units of microsiemens's */
573  pnd = sec->pnode;
574  nde = pnd[0]->extnode;
575  area = NODEAREA(sec->parentnode);
576  /* param[4] is rall_branch */
577  for (k = 0; k < nlayer; ++k) {
578  nde->_a[k] = -1.e2 * sec->prop->dparam[4].get<double>() / (nde->_b[k] * area);
579  }
580  for (j = 1; j < sec->nnode; j++) {
581  nde = pnd[j]->extnode;
582  area = NODEAREA(pnd[j - 1]);
583  for (k = 0; k < nlayer; ++k) {
584  nde->_a[k] = -1.e2 / (nde->_b[k] * area);
585  }
586  }
587  }
588  }
589  /* now the effect of parent on node equation. */
590  // ForAllSections(sec)
591  ITERATE(qsec, section_list) {
