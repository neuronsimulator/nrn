89 template <typename T>
90 void mech_data_layout_transform(T* data, int cnt, const std::vector<int>& array_dims, int layout) {
91  if (layout == Layout::AoS) {
92  throw std::runtime_error("AoS memory layout not implemented.");
93  }
94 
95  int n_vars = array_dims.size();
96  int row_width = std::accumulate(array_dims.begin(), array_dims.end(), 0);
97  int padded_cnt = nrn_soa_padded_size(cnt, layout);
98 
99  std::vector<T> tmp(padded_cnt * row_width);
100  std::copy(data, data + cnt * row_width, tmp.begin());
101 
102  size_t offset_var = 0;
103  for (size_t i_var = 0; i_var < n_vars; ++i_var) {
104  size_t K = array_dims[i_var];
105 
106  for (size_t i = 0; i < cnt; ++i) {
107  for (size_t k = 0; k < K; ++k) {
108  size_t i_dst = padded_cnt * offset_var + i * K + k;
109  size_t i_src = i * row_width + offset_var + k;
110  data[i_dst] = tmp[i_src];
111  }
112  }
113 
114  offset_var += K;
115  }
116 }

267  for (int i = 0; i < nt.n_presyn; ++i) {
268  preSynConditionEventFlags.push_back(F.read_int());
269  }
270 
271  nrn_assert(F.read_int() == -1);
272  restore_events(F);
273

547 void Phase2::restore_events(FileHandler& F) {
548  int type;
549  while ((type = F.read_int()) != 0) {
550  double time;
551  F.read_array(&time, 1);
552  switch (type) {
553  case NetConType: {
554  auto event = std::make_shared<NetConType_>();
555  event->time = time;
556  event->netcon_index = F.read_int();
557  events.emplace_back(type, event);
558  break;
559  }
560  case SelfEventType: {
561  auto event = std::make_shared<SelfEventType_>();
562  event->time = time;
563  event->target_type = F.read_int();
564  event->point_proc_instance = F.read_int();
565  event->target_instance = F.read_int();
566  F.read_array(&event->flag, 1);
567  event->movable = F.read_int();
568  event->weight_index = F.read_int();
569  events.emplace_back(type, event);
570  break;
571  }
572  case PreSynType: {
573  auto event = std::make_shared<PreSynType_>();
574  event->time = time;
575  event->presyn_index = F.read_int();
576  events.emplace_back(type, event);
577  break;
578  }
579  case NetParEventType: {
580  auto event = std::make_shared<NetParEvent_>();
581  event->time = time;
582  events.emplace_back(type, event);
583  break;
584  }
585  case PlayRecordEventType: {
586  auto event = std::make_shared<PlayRecordEventType_>();

638 void Phase2::pdata_relocation(const NrnThread& nt, const std::vector<Memb_func>& memb_func) {
639  // Some pdata may index into data which has been reordered from AoS to
640  // SoA. The four possibilities are if semantics is -1 (area), -5 (pointer),
641  // -9 (diam), // or 0-999 (ion variables).
642  // Note that pdata has a layout and the // type block in nt.data into which
643  // it indexes, has a layout.
644 
645  // For faster search of tmls[i].type == type, use a map.
646  // (perhaps would be better to replace tmls so that we can use tmls[type].
647  std::map<int, size_t> type2itml;
648  for (size_t i = 0; i < tmls.size(); ++i) {
649  if (tmls[i].pointer2type.size()) {
650  type2itml[tmls[i].type] = i;
651  }
652  }
653 
654  for (auto tml = nt.tml; tml; tml = tml->next) {
655  int type = tml->index;
656  int layout = corenrn.get_mech_data_layout()[type];
657  int* pdata = tml->ml->pdata;
658  int cnt = tml->ml->nodecount;
659  int szdp = corenrn.get_prop_dparam_size()[type];
660  int* semantics = memb_func[type].dparam_semantics;
661 
662  // compute only for ARTIFICIAL_CELL (has useful area pointer with semantics=-1)
663  if (!corenrn.get_is_artificial()[type]) {
664  if (szdp) {
665  if (!semantics)
666  continue; // temporary for HDFReport, Binreport which will be skipped in
667  // bbcore_write of HBPNeuron
668  nrn_assert(semantics);
669  }
670 
671  for (int i = 0; i < szdp; ++i) {
672  int s = semantics[i];
673  switch (s) {
674  case -1: // area
675  transform_int_data(
676  nt._actual_area - nt._data, cnt, pdata, i, szdp, layout, nt.end);
677  break;
678  case -9: // diam
679  transform_int_data(
680  nt._actual_diam - nt._data, cnt, pdata, i, szdp, layout, nt.end);
681  break;
682  case -5: // pointer assumes a pointer to membrane voltage
683  // or mechanism data in this thread. The value of the
684  // pointer on the NEURON side was analyzed by
685  // nrn_dblpntr2nrncore which returned the
686  // mechanism index and type. At this moment the index
687  // is in pdata and the type is in tmls[type].pointer2type.
688  // However the latter order is according to the nested
689  // iteration for nodecount { for szdp {}}
690  // Also the nodecount POINTER instances of mechanism
691  // might possibly point to differnt range variables.
692  // Therefore it is not possible to use transform_int_data
693  // and the transform must be done one at a time.
694  // So we do nothing here and separately iterate
695  // after this loop instead of the former voltage only
696  /**
697  transform_int_data(
698  nt._actual_v - nt._data, cnt, pdata, i, szdp, layout, nt.end);
699  **/
700  break;
701  default:
702  if (s >= 0 && s < 1000) { // ion
703  int etype = s;
704  /* if ion is SoA, must recalculate pdata values */
705  /* if ion is AoS, have to deal with offset */
706  Memb_list* eml = nt._ml_list[etype];
707  int edata0 = eml->data - nt._data;
708  int ecnt = eml->nodecount;
709  int ecnt_padded = nrn_soa_padded_size(ecnt, Layout::SoA);
710  int esz = corenrn.get_prop_param_size()[etype];
711  const std::vector<int>& array_dims = corenrn.get_array_dims()[etype];
712  for (int iml = 0; iml < cnt; ++iml) {
713  int* pd = pdata + nrn_i_layout(iml, cnt, i, szdp, layout);
714  int legacy_index = *pd; // relative to the ion data
715  nrn_assert((legacy_index >= 0) && (legacy_index < ecnt * esz));
716 
717  auto soaos_index = legacy2soaos_index(legacy_index, array_dims);
718  *pd = edata0 +
719  soaos2cnrn_index(soaos_index, array_dims, ecnt_padded, nullptr);
720  }
721  }
722  }
723  }
724  // Handle case -5 POINTER transformation (see comment above)
725  auto search = type2itml.find(type);
726  if (search != type2itml.end()) {
727  auto& ptypes = tmls[type2itml[type]].pointer2type;
728  assert(ptypes.size());
729  size_t iptype = 0;
730  for (int iml = 0; iml < cnt; ++iml) {
731  for (int i = 0; i < szdp; ++i) {
732  if (semantics[i] == -5) { // POINTER
733  int* pd = pdata + nrn_i_layout(iml, cnt, i, szdp, layout);
734  int ix = *pd; // relative to elem0
735  int ptype = ptypes[iptype++];
736  if (ptype == voltage) {
737  nrn_assert((ix >= 0) && (ix < nt.end));
738  int elem0 = nt._actual_v - nt._data;
739  *pd = elem0 + ix;
740  } else {
741  Memb_list* pml = nt._ml_list[ptype];
742  int pcnt = pml->nodecount;
743  int pcnt_padded = nrn_soa_padded_size(pcnt, Layout::SoA);
744  int psz = corenrn.get_prop_param_size()[ptype];
745  const std::vector<int>& array_dims =
746  corenrn.get_array_dims()[ptype];
747  nrn_assert((ix >= 0) && (ix < pcnt * psz));
748 
749  int elem0 = pml->data - nt._data;
750  auto soaos_index = legacy2soaos_index(ix, array_dims);
751  *pd =
752  elem0 +
753  soaos2cnrn_index(soaos_index, array_dims, pcnt_padded, nullptr);
754  }
755  }
756  }
757  }
758  ptypes.clear();
759  }
760  }

764 void Phase2::set_dependencies(const NrnThread& nt, const std::vector<Memb_func>& memb_func) {
765  /* here we setup the mechanism dependencies. if there is a mechanism dependency
766  * then we allocate an array for tml->dependencies otherwise set it to nullptr.
767  * In order to find out the "real" dependencies i.e. dependent mechanism
768  * exist at the same compartment, we compare the nodeindices of mechanisms
769  * returned by nrn_mech_depend.
770  */
771 
772  /* temporary array for dependencies */
773  int* mech_deps = (int*) ecalloc(memb_func.size(), sizeof(int));
774 
775  for (auto tml = nt.tml; tml; tml = tml->next) {
776  /* initialize to null */
777  tml->dependencies = nullptr;
778  tml->ndependencies = 0;
779 
780  /* get dependencies from the models */
781  int deps_cnt = nrn_mech_depend(tml->index, mech_deps);
782 
783  /* if dependencies, setup dependency array */
784  if (deps_cnt) {
785  /* store "real" dependencies in the vector */
786  std::vector<int> actual_mech_deps;
787 
788  Memb_list* ml = tml->ml;
789  int* nodeindices = ml->nodeindices;
790 
791  /* iterate over dependencies */
792  for (int j = 0; j < deps_cnt; j++) {
793  /* memb_list of dependency mechanism */
794  Memb_list* dml = nt._ml_list[mech_deps[j]];
795 
796  /* dependency mechanism may not exist in the model */
797  if (!dml)
798  continue;
799 
800  /* take nodeindices for comparison */
801  int* dnodeindices = dml->nodeindices;
802 
803  /* set_intersection function needs temp vector to push the common values */
804  std::vector<int> node_intersection;
805 
806  /* make sure they have non-zero nodes and find their intersection */
807  if ((ml->nodecount > 0) && (dml->nodecount > 0)) {
808  std::set_intersection(nodeindices,
809  nodeindices + ml->nodecount,
810  dnodeindices,
811  dnodeindices + dml->nodecount,
812  std::back_inserter(node_intersection));
813  }
814 
815  /* if they intersect in the nodeindices, it's real dependency */
816  if (!node_intersection.empty()) {
817  actual_mech_deps.push_back(mech_deps[j]);
818  }
819  }
820 
821  /* copy actual_mech_deps to dependencies */
822  if (!actual_mech_deps.empty()) {
823  tml->ndependencies = actual_mech_deps.size();
824  tml->dependencies = (int*) ecalloc(actual_mech_deps.size(), sizeof(int));
825  std::copy(actual_mech_deps.begin(), actual_mech_deps.end(), tml->dependencies);
826  }
827  }
828  }
829 
830  /* free temp dependency array */
831  free(mech_deps);
832 }
