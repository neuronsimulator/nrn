140  void destroy() {
141  if (!m_ptr) {
142  // Nothing to be done.
143  return;
144  }
145 
146  assert(m_data_ptr);
147  auto& data_container = *m_data_ptr;
148 
149  // We should still be a valid reference at this point.
150  assert(m_ptr);
151  assert(m_ptr.current_row() < data_container.size());
152 
153  // Prove that the bookkeeping works.
154  assert(data_container.at(m_ptr.current_row()) == m_ptr);
155 
156  bool terminate = false;
157  // Delete the corresponding row from `data_container`
158  try {
159  data_container.erase(m_ptr.current_row());
160  } catch (std::exception const& e) {
161  // Cannot throw from unique_ptr release/reset/destructor, this
162  // is the best we can do. Most likely what has happened is
163  // something like:
