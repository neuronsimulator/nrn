1045  std::unordered_map<PreSyn*, std::vector<size_t>> presyn2intdata;
1046  std::unordered_map<double*, std::vector<size_t>> weight2intdata;

1058  // factoring out of the loop bodies into set_info.)

1066  set_info(tqi, tid, core_te, netcon2intdata, presyn2intdata, weight2intdata);

1081  }

1091  for (auto iloc: iter->second) {
1092  core_te->intdata[iloc] = i;
1093  }
1094  presyn2intdata[ps][0] = i + NRN_SENTINAL;
1095  }
1096  }
1097  // all presyn2intdata should have been visited so all
1098  // presyn2intdata[ps][0] must be >= NRN_SENTINAL

1100  assert(iter.second[0] >= NRN_SENTINAL);
