217 /**
218  * @brief Type- and array-aware version of hoc_register_prop_size.
219  *
220  * hoc_register_prop_size did not propagate enough information to know which parts of the "data"
221  * size were ranges corresponding to a single array variable. This also aims to be ready for
222  * supporting multiple variable data types in MOD files.
223  */
224 // This is declared static on purpose. When we are using binary Python wheels and a user runs
225 // nrnivmodl on their local machine then we link together a libnrniv.so from the binary wheel
226 // with object files that were produced from .mod files on the user's machine. If the files
227 // compiled on the user's machine have the same number/type of variables as a built in mechanism
228 // that is embedded in libnrniv.so then there will be two calls to this function with the same
229 // template arguments. If this function was not static, those would have external linkage and a
230 // call from one of the user's .mod files might use a definition of this function from
231 // libnrniv.so. This can cause problems due to ABI mismatches, and the ::detail version above
232 // essentially exists to ensure that *it* is the interface between libnrniv.so and code compiled
233 // on the user's machine. Yes, this is horrible. See #1963 and #2234 for more information.
234 template <typename... Fields>
235 static void register_data_fields(int mech_type, Fields const&... fields) {
236  // Use of const char* aims to avoid wheel ABI issues with std::string
237  std::vector<std::pair<const char*, int>> param_info{};
238  std::vector<std::pair<const char*, const char*>> dparam_info{};
239  auto const process = [&](auto const& field) {
240  using field_t = std::decay_t<decltype(field)>;
241  using data_t = typename field_t::type;
242  if constexpr (std::is_same_v<data_t, double>) {
243  assert(field.semantics.empty());
244  param_info.emplace_back(field.name.c_str(), field.array_size);
245  } else {
246  static_assert(std::is_same_v<data_t, int> /* TODO */ || std::is_pointer_v<data_t>,
247  "only pointers, doubles and ints are supported");
248  assert(field.array_size == 1); // only scalar dparam data is supported
249  dparam_info.emplace_back(field.name.c_str(), field.semantics.c_str());
250  }
251  };
252  // fold expression with the comma operator is neater, but hits AppleClang expression depth
253  // limits for large sizeof...(Fields); the old initializer_list trick avoids that.
254  static_cast<void>(std::initializer_list<int>{(static_cast<void>(process(fields)), 0)...});
255  // beware, the next call crosses from translated mechanism code into the main NEURON library
256  detail::register_data_fields(mech_type, param_info, dparam_info);
257 }
