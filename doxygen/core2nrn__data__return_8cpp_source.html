53 /** @brief permuted array copied to unpermuted array
54  * If permute is NULL then just a copy
55  */
56 static void inverse_permute_copy(size_t n, double* permuted_src, double* dest, int* permute) {
57  if (permute) {
58  for (size_t i = 0; i < n; ++i) {
59  dest[i] = permuted_src[permute[i]];
60  }
61  } else {
62  std::copy(permuted_src, permuted_src + n, dest);
63  }
64 }

66 // See `soaos_copy_cnrn2nrn` for documentation.
67 template <bool needs_permute>
68 static void soaos_copy_cnrn2nrn_impl(size_t n,
69  int stride,
70  double const* const src,
71  std::vector<double*>& dest,
72  const std::vector<int>& array_dims,
73  int* permute) {
74  // i : runs over instances: 0, ..., n.
75  // j : runs over variables: 0, ..., array_dims.size() =: n_vars.
76  // k : runs over array dimension: 0, ..., array_dims[i_var] =: K.
77 
78  int n_vars = array_dims.size();
79  double const* src_var = src;
80  for (size_t j_var = 0; j_var < n_vars; ++j_var) {
81  size_t K = array_dims[j_var];
82 
83  for (size_t i = 0; i < n; ++i) {
84  size_t i_src = needs_permute ? static_cast<size_t>(permute[i]) : i;
85  for (size_t k = 0; k < K; ++k) {
86  dest[j_var][i * K + k] = src_var[i_src * K + k];
87  }
88  }
89 
90  src_var += stride * K;
91  }
92 }

94 /** @brief SoAoS unpermuted mechanism data copied to unpermuted SoAoS data.
95  *
96  * This function copies the CoreNEURON SoAoS format into the NEURON format. There
97  * are two differences:
98  * 1. NRN allocates a separate array for each variable, while CoreNEURON
99  * allocates a single large chunk of memory.
100  * 2. CoreNEURON pads the instance number.
101  *
102  * For ARRAY variables both NRN and CoreNRN store consecutive elements of the
103  * array consecutively in memory.
104  *
105  * Let `K` be the number of variables, counting array variables as one
106  * variable. Then `dest` is an array of `K` pointers, one for each variable of
107  * the mechanism. The pointer `dest[i]` is an array of size `n*array_dims[i]`
108  * and the array element `k` for instance `i` is stored at `dest[i*n + k]`.
109  *
110  * The pointer `src` points to `stride*sum(array_dims)` doubles.
111  *
112  * If `permute != nullptr`, then then the instance is permuted, i.e. the
113  * instance index is `permute[i]` in `src` (and `i` in `dest`). If `permute ==
114  * nullptr` no permutation is performed.
115  */
116 static void soaos_copy_cnrn2nrn(size_t n,
117  int stride,
118  double const* const src,
119  std::vector<double*>& dest,
120  const std::vector<int>& array_dims,
121  int* permute) {
122  if (permute == nullptr) {
123  soaos_copy_cnrn2nrn_impl<false>(n, stride, src, dest, array_dims, permute);
124  } else {
125  soaos_copy_cnrn2nrn_impl<true>(n, stride, src, dest, array_dims, permute);
126  }
127 }

129 /** @brief AoS mechanism data copied to AoS data.
130  * dest is an array of n pointers to the beginning of each sz length array.
131  * src is a contiguous array of n segments of size sz.
132  */
133 static void aos2aos_copy(size_t n, int sz, double* src, std::vector<double*>& dest) {
134  for (size_t instance = 0; instance < n; ++instance) {
135  double* s = src + (instance * sz);
136  for (auto i = 0; i < sz; ++i) {
137  dest[i][instance] = s[i];
138  }
139  }

144 static void core2nrn_corepointer(int tid, NrnThreadMembList* tml) {
145  // Based on get_bbcore_write fragment in nrn_checkpoint.cpp
146  int type = tml->index;
147  if (!corenrn.get_bbcore_write()[type]) {
148  return;
149  }
150  NrnThread& nt = nrn_threads[tid];
151  Memb_list* ml = tml->ml;
152  double* d = nullptr;
153  Datum* pd = nullptr;
154  int layout = corenrn.get_mech_data_layout()[type];
155  int dsz = corenrn.get_prop_param_size()[type];
156  int pdsz = corenrn.get_prop_dparam_size()[type];
157  int aln_cntml = nrn_soa_padded_size(ml->nodecount, layout);
158 
159  int icnt = 0;
160  int dcnt = 0;
161  // data size and allocate
162  for (int j = 0; j < ml->nodecount; ++j) {
163  int jp = j;
164  if (ml->_permute) {
165  jp = ml->_permute[j];
166  }
167  d = ml->data + nrn_i_layout(jp, ml->nodecount, 0, dsz, layout);
168  pd = ml->pdata + nrn_i_layout(jp, ml->nodecount, 0, pdsz, layout);
169  (*corenrn.get_bbcore_write()[type])(
170  nullptr, nullptr, &dcnt, &icnt, 0, aln_cntml, d, pd, ml->_thread, &nt, ml, 0.0);
171  }
172 
173  std::unique_ptr<int[]> iArray;
174  std::unique_ptr<double[]> dArray;
175  if (icnt) {
176  iArray.reset(new int[icnt]);
177  }
178  if (dcnt) {
179  dArray.reset(new double[dcnt]);
180  }
181  icnt = dcnt = 0;
182  for (int j = 0; j < ml->nodecount; j++) {
183  int jp = j;
184 
185  if (ml->_permute) {
186  jp = ml->_permute[j];
187  }
188 
189  d = ml->data + nrn_i_layout(jp, ml->nodecount, 0, dsz, layout);
190  pd = ml->pdata + nrn_i_layout(jp, ml->nodecount, 0, pdsz, layout);
191 
192  (*corenrn.get_bbcore_write()[type])(dArray.get(),
193  iArray.get(),
194  &dcnt,
195  &icnt,
196  0,
197  aln_cntml,
198  d,
199  pd,
200  ml->_thread,
201  &nt,
202  ml,
203  0.0);
204  }

301 /** @brief copy data back to NEURON.
302  * Copies t, voltage, i_membrane_ if it used, and mechanism param data.
303  * Copies event queue and related state, e.g. WATCH, VecPlayContinuous.
304  */
305 void core2nrn_data_return() {
306  if (!nrn2core_type_return_) {
307  return;
308  }
309 
310  (*core2nrn_clear_queues_)(nrn_threads[0]._t); // all threads at same time
311 
312  for (int tid = 0; tid < nrn_nthread; ++tid) {
313  size_t n = 0;
314  double* data = nullptr;
315  NrnThread& nt = nrn_threads[tid];
316  std::vector<double*> mdata{};
317  n = (*nrn2core_type_return_)(0, tid, data, mdata); // 0 means time
318  if (n) { // not the empty thread
319  data[0] = nt._t;
320  }
321 
322  if (nt.end) { // transfer voltage and possibly i_membrane_
323  n = (*nrn2core_type_return_)(voltage, tid, data, mdata);
324  assert(n == size_t(nt.end) && data);
325  inverse_permute_copy(n, nt._actual_v, data, nt._permute);
326 
327  if (nt.nrn_fast_imem) {
328  n = (*nrn2core_type_return_)(i_membrane_, tid, data, mdata);
329  assert(n == size_t(nt.end) && data);
330  inverse_permute_copy(n, nt.nrn_fast_imem->nrn_sav_rhs, data, nt._permute);
331  }
332  }
333 
334  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
335  int mtype = tml->index;
336  Memb_list* ml = tml->ml;
337  n = (*nrn2core_type_return_)(mtype, tid, data, mdata);
338  assert(n == size_t(ml->nodecount) && !mdata.empty());
339  if (n == 0) {
340  continue;
341  }
342  // NEURON is AoS, CoreNEURON may be SoA and may be permuted.
343  // On the NEURON side, the data is actually contiguous because of
344  // cache_efficient, but that may not be the case for ARTIFICIAL_CELL.
345  // For initial implementation simplicity, use the mdata info which gives
346  // a double* for each param_size mech instance.
347  int* permute = ml->_permute;
348  double* cndat = ml->data;
349  int layout = corenrn.get_mech_data_layout()[mtype];
350  int sz = corenrn.get_prop_param_size()[mtype];
351  const std::vector<int>& array_dims = corenrn.get_array_dims()[mtype];
352  if (layout == Layout::SoA) {
353  int stride = ml->_nodecount_padded;
354  soaos_copy_cnrn2nrn(n, stride, cndat, mdata, array_dims, permute);
355  } else { /* AoS */
356  aos2aos_copy(n, sz, cndat, mdata);
357  }
358 
359  core2nrn_corepointer(tid, tml);
360  c2n_nmodlrandom(tid, tml);
361  }
362 
363  // Copy the event queue and related state.
364  core2nrn_tqueue(nt);
365  }
366  core2nrn_vecplay();
367  core2nrn_watch();
368 }

370 /** @brief Callbacks into NEURON for WatchCondition.
371  */
372 static void core2nrn_watch() {
373  (*core2nrn_watch_clear_)();
374 
375  // much of the following nested iterations follows the
376  // watch_activate_clear() function in sim/finitialize.cpp, though here
377  // we iterate over nt._watch_types instead of nt.tml and then picking out
378  // the WATCH relevant types with corenrn.get_watch_check().
379  for (int tid = 0; tid < nrn_nthread; ++tid) {
380  NrnThread& nt = nrn_threads[tid];
381  if (nt._watch_types) {
382  for (int i = 0; nt._watch_types[i] != 0; ++i) {
383  int type = nt._watch_types[i];
384  Memb_list& ml = *(nt._ml_list[type]);
385  int nodecount = ml.nodecount;
386  Core2NrnWatchInfo watch_info(ml.nodecount);
387  int* permute = ml._permute;
388  int* pdata = (int*) ml.pdata;
389  int dparam_size = corenrn.get_prop_dparam_size()[type];
390  int layout = corenrn.get_mech_data_layout()[type];
391  int first, last;
392  watch_datum_indices(type, first, last);
393  int watch_begin = first;
394  for (int iml = 0; iml < nodecount; ++iml) {
395  int iml_permute = permute ? permute[iml] : iml;
396  Core2NrnWatchInfoItem& wiv = watch_info[iml];
397  for (int ix = first; ix <= last; ++ix) {
398  int datum =
399  pdata[nrn_i_layout(iml_permute, nodecount, ix, dparam_size, layout)];
400  if (datum & 2) { // activated
401  bool above_thresh = bool(datum & 1);
402  wiv.push_back(std::pair<int, bool>(ix, above_thresh));
403  }
404  }
405  }
406  (*core2nrn_watch_activate_)(tid, type, watch_begin, watch_info);
407  }
408  }
409  }

527 // return false unless q is pushed to sewm
528 static bool core2nrn_tqueue_item(TQItem* q, SelfEventWeightMap& sewm, NrnThread& nt) {
529  DiscreteEvent* d = (DiscreteEvent*) q->data_;
530  double td = q->t_;
531  bool in_sewm = false;
532 
533  switch (d->type()) {
534  case NetConType: {
535  NetCon* nc = (NetCon*) d;
536  assert(nc >= nt.netcons && (nc < (nt.netcons + nt.n_netcon)));
537  size_t nc_index = nc - nt.netcons;
538  (*core2nrn_NetCon_event_)(nt.id, td, nc_index);
539  break;
540  }
541  case SelfEventType: {
542  SelfEvent* se = (SelfEvent*) d;
543  Point_process* pnt = se->target_;
544  assert(pnt->_tid == nt.id);
545  int tar_type = (int) pnt->_type;
546  Memb_list* ml = nt._ml_list[tar_type];
547  if (ml->_permute) { // if permutation, then make inverse available
548  // Doing this here because we don't know, in general, which
549  // mechanisms use SelfEvent
550  if (type2invperm.count(tar_type) == 0) {
551  type2invperm[tar_type] = inverse_permute(ml->_permute, ml->nodecount);
552  }
553  }
554  double flag = se->flag_;
555  TQItem** movable = (TQItem**) (se->movable_);
556  int is_movable = (movable && *movable == q) ? 1 : 0;
557  int weight_index = se->weight_index_;
558  // the weight_index is useless on the NEURON side so we need
559  // to convert that to NetCon index and let the NEURON side
560  // figure out the weight_index. To figure out the netcon_index
561  // construct a {weight_index : [TQItem]} here for any
562  // weight_index >= 0, otherwise send it NEURON now.
563  if (weight_index >= 0) {
564  // Potentially several SelfEvent TQItem* associated with
565  // same weight index. More importantly, collect them all
566  // so that we only need to iterate over the nt.netcons once
567  sewm[weight_index].push_back(q);
568  in_sewm = true;
569 
570  } else {
571  int tar_index = pnt->_i_instance; // correct for no permutation
572  if (ml->_permute) {
573  tar_index = type2invperm[tar_type][tar_index];
574  }
575  (*core2nrn_SelfEvent_event_noweight_)(nt.id, td, tar_type, tar_index, flag, is_movable);
576  delete se;
577  }
578  break;
579  }
580  case PreSynType: {
581  // nothing to transfer
582  // `d` can be cast to PreSyn*
583  break;
584  }
585  case NetParEventType: {
586  // nothing to transfer
587  break;
588  }
589  case PlayRecordEventType: {
590  // nothing to transfer
591  break;
592  }
593  case ReportEventType: {
594  // no need to transfer ReportEvent
595  break;
596  }
597  default: {
598  // In particular, InputPreSyn does not appear in tqueue as it
599  // immediately fans out to NetCon.
600  std::stringstream qetype;
601  qetype << d->type();
602  hoc_execerror("core2nrn_tqueue_item -> unimplemented queue event type:",
603  qetype.str().c_str());
604  break;
605  }
606  }
607  return in_sewm;
608 }

610 void core2nrn_tqueue(NrnThread& nt) {
611  // VecPlayContinuous
612 
613  // PatternStim
614 
615  // nrn_checkpoint.cpp has:
616  // Avoid extra spikes due to some presyn voltages above threshold
617 
618  // PreSyn.flag_ that are on
619  core2nrn_PreSyn_flag(nt);
620 
621  // The items on the queue
622  NetCvodeThreadData& ntd = net_cvode_instance->p[nt.id];
623  // make sure all buffered interthread events are on the queue
624  ntd.enqueue(net_cvode_instance, &nt);
625 
626  TQueue<QTYPE>* tqe = ntd.tqe_;
627  TQItem* q;
628  SelfEventWeightMap sewm;
629  // TQItems from atomic_dq
630  while ((q = tqe->atomic_dq(1e20)) != nullptr) {
631  if (core2nrn_tqueue_item(q, sewm, nt) == false) {
632  delete q;
633  }
634  }
635  // TQitems from binq_
636  for (q = tqe->binq_->first(); q; q = tqe->binq_->next(q)) {
637  bool const result = core2nrn_tqueue_item(q, sewm, nt);
638  assert(result == false);
639  }
640 
641  // For self events with weight, find the NetCon index and send that
642  // to NEURON.
643  // If the SelfEventWeightMap approach (and the corresponding pattern
644  // on the nrn2core side in NEURON) ends up being too expensive in space
645  // or time, it would be possible to modify SelfEvent to use the NetCon
646  // index instead of the weight index, and then directly determine the
647  // NetCon within the core2nrn_tqueue_item function above and call
648  // (*core2nrn_SelfEvent_event_) from there.
649  if (!sewm.empty()) {
650  for (int nc_index = 0; nc_index < nt.n_netcon; ++nc_index) {
651  NetCon& nc = nt.netcons[nc_index];
652  int weight_index = nc.u.weight_index_;
653  auto search = sewm.find(weight_index);
654  if (search != sewm.end()) {
655  const auto& tqitems = search->second;
656  for (auto q: tqitems) {
657  DiscreteEvent* d = (DiscreteEvent*) (q->data_);
658  double td = q->t_;
659  assert(d->type() == SelfEventType);
660  SelfEvent* se = (SelfEvent*) d;
661  int tar_type = se->target_->_type;
662  // Note that instead of getting tar_index from the permuted
663  // pnt->_i_instance here and for the noweight case above
664  // which then needs the possibly large inverse permutation
665  // vectors, it would save some space to use the unpermuted
666  // nt.pntprocs array along with a much shorter vector
667  // of type offsets.
668  int tar_index = se->target_->_i_instance;
669  if (nt._ml_list[tar_type]->_permute) {
670  tar_index = type2invperm[tar_type][tar_index];
671  }
672  double flag = se->flag_;
673  TQItem** movable = (TQItem**) (se->movable_);
674  int is_movable = (movable && *movable == q) ? 1 : 0;
675  (*core2nrn_SelfEvent_event_)(
676  nt.id, td, tar_type, tar_index, flag, nc_index, is_movable);
677  delete q;
678  delete se;
679  }
680  }
681  }
682  }
683 
684  clear_inv_perm_for_selfevent_targets();
685 }
