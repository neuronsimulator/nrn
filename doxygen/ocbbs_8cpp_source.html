1149 // This helper execute depending of the style of the message.
1150 // style == 0:
1151 // This is a string that is a hoc statement, execute it with `hoc_obj_run`.
1152 // Return `nullptr`, set `size` argument to 0.
1153 // Not listen to `exec`.
1154 // No arguments.
1155 // style == 1:
1156 // This is a function name as a string followed by a list of arguments.
1157 // Return `nullptr` and set `size` to 0.
1158 // Dry run if `exec` is false.
1159 // style == 2:
1160 // From a template, find a object with an id and inside this object find a member
1161 // with a name as a string. Execute this member with arguments.
1162 // Return `nullptr` and set `size` to 0.
1163 // Dry run if `exec` is false.
1164 // style == 3:
1165 // A python pickle (https://docs.python.org/3/library/pickle.html) followed by arguments.
1166 // Return a string that is of size `size`.
1167 // Dry run if `exec` is false.
1168 std::vector<char> BBSImpl::execute_helper(int id, bool exec) {
1169  int subworld = (nrnmpi_numprocs > 1 && nrnmpi_numprocs_bbs < nrnmpi_numprocs_world);
1170  int style = upkint();
1171  if (subworld) {
1172  assert(nrnmpi_myid == 0);
1173  int info[2];
1174  info[0] = id;
1175  info[1] = style;
1176  nrnmpi_int_broadcast(info, 2, 0);
1177  }
1178  std::vector<char> rs{};
1179  switch (style) {
1180  case 0: {
1181  char* statement = upkstr();
1182  if (subworld) {
1183  int size = strlen(statement) + 1;
1184  nrnmpi_int_broadcast(&size, 1, 0);
1185  nrnmpi_char_broadcast(statement, size, 0);
1186  }
1187  hoc_obj_run(statement, nullptr);
1188  delete[] statement;
1189  } break;
1190  default: {
1191  std::vector<char> python_pickle{}; // Only for style == 3
1192  Symbol* fname = nullptr;
1193  Object* ob = nullptr;
1194  std::list<char*> sarg; // Store the strings pointer to delete[] them later
1195  // Use a list because, we push pointers of the object into
1196  // the hoc stack
1197  int narg = 0; // total number of args
1198  if (style == 2) { // object first
1199  char* template_name = upkstr();
1200  int object_index = upkint(); // object index
1201  Symbol* sym = hoc_lookup(template_name);
1202  if (sym) {
1203  sym = hoc_which_template(sym);
1204  }
1205  if (!sym) {
1206  hoc_execerror(template_name, "is not a template");
1207  }
1208  hoc_Item *q, *ql;
1209  ql = sym->u.ctemplate->olist;
1210  ITERATE(q, ql) {
1211  ob = OBJ(q);
1212  if (ob->index == object_index) {
1213  break;
1214  }
1215  ob = nullptr;
1216  }
1217  if (!ob) {
1218  fprintf(stderr,
1219  "%s[%d] is not an Object in this process\n",
1220  template_name,
1221  object_index);
1222  hoc_execerror("ParallelContext execution error", nullptr);
1223  }
1224  delete[] template_name;

1226  fname = hoc_table_lookup(fname_str, sym->u.ctemplate->symtable);
1227  if (!fname) {

1229  hoc_execerror("ParallelContext execution error", nullptr);
1230  }

1233  hoc_execerror("with subworlds, this submit style not implemented", nullptr);
1234  }

1236  python_pickle = upkpickle();
1237  if (subworld) {
1238  int size = static_cast<int>(python_pickle.size());

1246  nrnmpi_int_broadcast(&size, 1, 0);
