75 uint32_t nrnRan4int(uint32_t* idx1, uint32_t idx2) {
76  uint32_t u, v, w, m, n;
77  /* 64-bit hash */
78  n = (*idx1)++;
79  m = idx2;
80 
81  w = n ^ 0xbaa96887;
82  v = w >> 16;
83  w &= 0xffff;
84  u = ~((v - w) * (v + w));
85  /*m ^= (((u >> 16) | (u << 16)) ^ 0xb4f0c4a7) + w * v;*/
86  m ^= (((u >> 16) | (u << 16)) ^ 0x4b0f3b58) + w * v;
87 
88  w = m ^ 0x1e17d32c;
89  v = w >> 16;
90  w &= 0xffff;
91  u = ~((v - w) * (v + w));
92  /*n ^= (((u >> 16) | (u << 16)) ^ 0x178b0f3c) + w * v;*/
93  n ^= (((u >> 16) | (u << 16)) ^ 0xe874f0c3) + w * v;
94  return n;
95 
96  w = n ^ 0x03bcdc3c;
97  v = w >> 16;
98  w &= 0xffff;
99  u = (v - w) * (v + w);
