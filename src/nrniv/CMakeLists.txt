macro(nocmodl_mod_to_c bname)
add_custom_command(
  OUTPUT ${bname}.c
  COMMAND
    nocmodl ${bname}.mod
  COMMAND sed "'s/_reg()/_reg_()/'" ${bname}.c > ${bname}.c.tmp
  COMMAND mv ${bname}.c.tmp ${bname}.c
  DEPENDS ${PROJECT_BINARY_DIR}/bin/nocmodl ${bname}.mod
)
endmacro()
